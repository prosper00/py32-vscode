98HAL_StatusTypeDef HAL_CRC_Init(CRC_HandleTypeDef *hcrc)
99{
100 /* Check the CRC handle allocation */
101 if (hcrc == NULL)
102 {
103 return HAL_ERROR;
104 }
105
106 /* Check the parameters */
107 assert_param(IS_CRC_ALL_INSTANCE(hcrc->Instance));
108
109 if (hcrc->State == HAL_CRC_STATE_RESET)
110 {
111 /* Allocate lock resource and initialize it */
112 hcrc->Lock = HAL_UNLOCKED;
113 /* Init the low level hardware */
114 HAL_CRC_MspInit(hcrc);
115 }
116
117 /* Change CRC peripheral state */
118 hcrc->State = HAL_CRC_STATE_READY;
119
120 /* Return function status */
121 return HAL_OK;
122}

129HAL_StatusTypeDef HAL_CRC_DeInit(CRC_HandleTypeDef *hcrc)
130{
131 /* Check the CRC handle allocation */
132 if (hcrc == NULL)
133 {
134 return HAL_ERROR;
135 }
136
137 /* Check the parameters */
138 assert_param(IS_CRC_ALL_INSTANCE(hcrc->Instance));
139
140 /* Check the CRC peripheral state */
141 if (hcrc->State == HAL_CRC_STATE_BUSY)
142 {
143 return HAL_BUSY;
144 }
145
146 /* Change CRC peripheral state */
147 hcrc->State = HAL_CRC_STATE_BUSY;
148
149 /* Reset CRC calculation unit */
150 __HAL_CRC_DR_RESET(hcrc);
151
152 /* Reset IDR register content */
153 CLEAR_BIT(hcrc->Instance->IDR, CRC_IDR_IDR);
154
155 /* DeInit the low level hardware */
156 HAL_CRC_MspDeInit(hcrc);
157
158 /* Change CRC peripheral state */
159 hcrc->State = HAL_CRC_STATE_RESET;
160
161 /* Process unlocked */
162 __HAL_UNLOCK(hcrc);
163
164 /* Return function status */
165 return HAL_OK;
166}
