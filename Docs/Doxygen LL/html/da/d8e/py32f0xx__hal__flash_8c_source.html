216HAL_StatusTypeDef FLASH_WaitForLastOperation(uint32_t Timeout)
217{
218 /* Wait for the FLASH operation to complete by polling on BUSY flag to be reset.
219 Even if the FLASH operation fails, the BUSY flag will be reset and an error
220 flag will be set */
221 uint32_t timeout = HAL_GetTick() + Timeout;
222
223 /* Wait if any operation is ongoing */
224 while (__HAL_FLASH_GET_FLAG(FLASH_FLAG_BSY) != 0x00U)
225 {
226 if (HAL_GetTick() >= timeout)
227 {
228 return HAL_TIMEOUT;
229 }
230 }
231
232 /* Clear SR register */
233 FLASH->SR = FLASH_FLAG_SR_CLEAR;
234
235 return HAL_OK;
236}

281static __RAM_FUNC void FLASH_Program_Page(uint32_t Address, uint32_t * DataAddress)
282{
283
284 uint8_t index=0;
285 uint32_t dest = Address;
286 uint32_t * src = DataAddress;
287 uint32_t primask_bit;
288
289 SET_BIT(FLASH->CR, FLASH_CR_PG);
290 /* Enter critical section */
291 primask_bit = __get_PRIMASK();
292 __disable_irq();
293 /* 32 words*/
294 while(index<32U)
295 {
296 *(uint32_t *)dest = *src;
297 src += 1U;
298 dest += 4U;
299 index++;
300 if(index==31)
301 {
302 SET_BIT(FLASH->CR, FLASH_CR_PGSTRT);
303 }
304 }
305
306 /* Exit critical section: restore previous priority mask */
307 __set_PRIMASK(primask_bit);
308}

325HAL_StatusTypeDef HAL_FLASH_Erase(FLASH_EraseInitTypeDef *pEraseInit, uint32_t *PageSectorError)
326{
327 HAL_StatusTypeDef status = HAL_ERROR;
328 uint32_t address = 0U;
329
330 /* Process Locked */
331 __HAL_LOCK(&pFlash);
332
333 /* Config flash timming */
334 __HAL_FLASH_TIMMING_SEQUENCE_CONFIG();
335
336 /* Check the parameters */
337 assert_param(IS_FLASH_TYPEERASE(pEraseInit->TypeErase));
338
339 if (pEraseInit->TypeErase == FLASH_TYPEERASE_MASSERASE)
340 {
341 /* Mass Erase requested for Bank1 */
342 /* Wait for last operation to be completed */
343 if (FLASH_WaitForLastOperation((uint32_t)FLASH_TIMEOUT_VALUE) == HAL_OK)
344 {
345 /*Mass erase to be done*/
346 FLASH_MassErase();
347
348 /* Wait for last operation to be completed */
349 status = FLASH_WaitForLastOperation((uint32_t)FLASH_TIMEOUT_VALUE);
350
351 /* If the erase operation is completed, disable the MER Bit */
352 CLEAR_BIT(FLASH->CR, FLASH_CR_MER);
353 }
354 }else if(pEraseInit->TypeErase == FLASH_TYPEERASE_PAGEERASE)
355 {
356 /* Page Erase is requested */
357 /* Check the parameters */
358 assert_param(IS_FLASH_PROGRAM_ADDRESS(pEraseInit->PageAddress));
359 assert_param(IS_FLASH_NB_PAGES(pEraseInit->PageAddress, pEraseInit->NbPages));
360
361 /* Wait for last operation to be completed */
362 if (FLASH_WaitForLastOperation((uint32_t)FLASH_TIMEOUT_VALUE) == HAL_OK)
363 {
364 /*Initialization of PageSectorError variable*/
365 *PageSectorError = 0xFFFFFFFFU;
366
367 /* Erase page by page to be done*/
368 for(address = pEraseInit->PageAddress;
369 address < ((pEraseInit->NbPages * FLASH_PAGE_SIZE) + pEraseInit->PageAddress);
370 address += FLASH_PAGE_SIZE)
371 {
372 FLASH_PageErase(address);
373
374 /* Wait for last operation to be completed */
375 status = FLASH_WaitForLastOperation((uint32_t)FLASH_TIMEOUT_VALUE);
376 /* If the erase operation is completed, disable the PER Bit */
377 CLEAR_BIT(FLASH->CR, FLASH_CR_PER);
378 if (status != HAL_OK)
379 {
380 /* In case of error, stop erase procedure and return the faulty address */
381 *PageSectorError = address;
382 break;
383 }
384 }
385 }
386 }
387 else if(pEraseInit->TypeErase == FLASH_TYPEERASE_SECTORERASE)
388 {
389 /* Sector Erase is requested */
390 /* Check the parameters */
391 assert_param(IS_FLASH_PROGRAM_ADDRESS(pEraseInit->SectorAddress));
392 assert_param(IS_FLASH_NB_SECTORS(pEraseInit->SectorAddress, pEraseInit->NbSectors));
393
394 /* Wait for last operation to be completed */
395 if (FLASH_WaitForLastOperation((uint32_t)FLASH_TIMEOUT_VALUE) == HAL_OK)
396 {
397 /*Initialization of PageSectorError variable*/
398 *PageSectorError = 0xFFFFFFFFU;
399
400 /* Erase page by page to be done*/
401 for(address = pEraseInit->SectorAddress;
402 address < ((pEraseInit->NbSectors * FLASH_SECTOR_SIZE) + pEraseInit->SectorAddress);
403 address += FLASH_SECTOR_SIZE)
404 {
405 FLASH_SectorErase(address);
406 /* Wait for last operation to be completed */
407 status = FLASH_WaitForLastOperation((uint32_t)FLASH_TIMEOUT_VALUE);
408 /* If the erase operation is completed, disable the SER Bit */
409 CLEAR_BIT(FLASH->CR, FLASH_CR_SER);
410 if (status != HAL_OK)
411 {
412 /* In case of error, stop erase procedure and return the faulty address */
413 *PageSectorError = address;
414 break;
415 }
416 }
417 }
418 }
419
420 /* Process Unlocked */
421 __HAL_UNLOCK(&pFlash);
422
423 return status;
424}

433HAL_StatusTypeDef HAL_FLASH_Erase_IT(FLASH_EraseInitTypeDef *pEraseInit)
434{
435 HAL_StatusTypeDef status;
436
437 /* Check the parameters */
438 assert_param(IS_FLASH_TYPEERASE(pEraseInit->TypeErase));
439
440 /* Process Locked */
441 __HAL_LOCK(&pFlash);
442
443 /* Config flash timming */
444 __HAL_FLASH_TIMMING_SEQUENCE_CONFIG();
445
446 /* Reset error code */
447 pFlash.ErrorCode = HAL_FLASH_ERROR_NONE;
448
449 /* save procedure for interrupt treatment */
450 pFlash.ProcedureOnGoing = pEraseInit->TypeErase;
451
452 /* Wait for last operation to be completed */
453 status = FLASH_WaitForLastOperation(FLASH_TIMEOUT_VALUE);
454
455 if (status != HAL_OK)
456 {
457 /* Process Unlocked */
458 __HAL_UNLOCK(&pFlash);
459 }
460 else
461 {
462 /* Enable End of Operation and Error interrupts */
463 FLASH->CR |= FLASH_CR_EOPIE | FLASH_CR_ERRIE;
464
465 if (pEraseInit->TypeErase == FLASH_TYPEERASE_MASSERASE)
466 {
467 /* Set Page to 0 for Interrupt callback management */
468 pFlash.PageOrSector = 0;
469
470 /* Proceed to Mass Erase */
471 FLASH_MassErase();
472 }else if (pEraseInit->TypeErase == FLASH_TYPEERASE_PAGEERASE)
473 {
474 /* Erase by page to be done */
475 pFlash.NbPagesSectorsToErase = pEraseInit->NbPages;
476 pFlash.PageOrSector = pEraseInit->PageAddress;
477
478 /*Erase 1st page and wait for IT */
479 FLASH_PageErase(pEraseInit->PageAddress);
480 }else if (pEraseInit->TypeErase == FLASH_TYPEERASE_SECTORERASE)
481 {
482 /* Erase by sector to be done */
483 pFlash.NbPagesSectorsToErase = pEraseInit->NbSectors;
484 pFlash.PageOrSector = pEraseInit->SectorAddress;
485
486 FLASH_SectorErase(pEraseInit->SectorAddress);
487 }
488 }
489
490 /* return status */
491 return status;

515HAL_StatusTypeDef HAL_FLASH_Program(uint32_t TypeProgram, uint32_t Address, uint32_t * DataAddr )
516{
517 HAL_StatusTypeDef status = HAL_ERROR;
518
519 /* Process Locked */
520 __HAL_LOCK(&pFlash);
521
522 /* Config flash timming */
523 __HAL_FLASH_TIMMING_SEQUENCE_CONFIG();
524
525 /* Check the parameters */
526 assert_param(IS_FLASH_TYPEPROGRAM(TypeProgram));
527 assert_param(IS_FLASH_PROGRAM_ADDRESS(Address));
528
529 /* must be the first address of the PAGE */
530 if(Address%FLASH_PAGE_SIZE)
531 {
532 return HAL_ERROR;
533 }
534
535 /* Wait for last operation to be completed */
536 status = FLASH_WaitForLastOperation(FLASH_TIMEOUT_VALUE);
537

539 {
540 FLASH_Program_Page(Address, DataAddr);
541
542 /* Wait for last operation to be completed */
543 status = FLASH_WaitForLastOperation(FLASH_TIMEOUT_VALUE);
544 /* If the program operation is completed, disable the PG Bit */
545 CLEAR_BIT(FLASH->CR, FLASH_CR_PG);
546 /* In case of error, stop programming procedure */
547 if (status != HAL_OK)
548 {
549 return status;
550 }
551 }
552
553 /* Process Unlocked */
554 __HAL_UNLOCK(&pFlash);
555
556 return status;

752HAL_StatusTypeDef HAL_FLASH_OBProgram(FLASH_OBProgramInitTypeDef *pOBInit)
753{
754 uint32_t optr;
755 HAL_StatusTypeDef status = HAL_ERROR;
756
757 /* Process Locked */
758 __HAL_LOCK(&pFlash);
759
760 /* Config flash timming */
761 __HAL_FLASH_TIMMING_SEQUENCE_CONFIG();
762
763 /* Check the parameters */
764 assert_param(IS_OPTIONBYTE(pOBInit->OptionType));
765
766 /* WRP register */
767 if ((pOBInit->OptionType & OPTIONBYTE_WRP) != 0)
768 {
769 /* Write protection configuration */
770 FLASH->WRPR = (uint16_t)(~(pOBInit->WRPSector));
771 }
772
773 /* SDK register */
774 if ((pOBInit->OptionType & OPTIONBYTE_SDK) != 0)
775 {
776 /* SDK protection configuration */
777 FLASH->SDKR = (pOBInit->SDKStartAddr) | (pOBInit->SDKEndAddr<<8);
778 }
779
780 /* Option register */
781 if ((pOBInit->OptionType & (OPTIONBYTE_RDP | OPTIONBYTE_USER)) == (OPTIONBYTE_RDP | OPTIONBYTE_USER))
782 {
783 /* Fully modify OPTR register with RDP & user data */
784 FLASH_OB_OptrConfig(pOBInit->USERType, pOBInit->USERConfig, pOBInit->RDPLevel);
785 }
786 else if((pOBInit->OptionType & OPTIONBYTE_RDP) != 0x00U)
787 {
788 /* Only modify RDP so get current user data */
789 optr = FLASH_OB_GetUser();
790 FLASH_OB_OptrConfig(optr, optr, pOBInit->RDPLevel);
791 }
792 else if ((pOBInit->OptionType & OPTIONBYTE_USER) != 0x00U)
793 {
794 /* Only modify user so get current RDP level */
795 optr = FLASH_OB_GetRDP();
796 FLASH_OB_OptrConfig(pOBInit->USERType, pOBInit->USERConfig, optr);
797 }
798 else
799 {
800 /* nothing to do */
801 }
802
803 /* starts to modify Flash Option bytes */
804 FLASH->CR|=FLASH_CR_OPTSTRT;
805
806 /* set bit EOPIE */
807 FLASH->CR|=FLASH_CR_EOPIE;
808
809 /* trigger program */
810 *((__IO uint32_t *)(0x40022080))=0xff;
811
812 /* Wait for last operation to be completed */
813 status = FLASH_WaitForLastOperation((uint32_t)FLASH_TIMEOUT_VALUE);
814
815 /* Process Unlocked */
816 __HAL_UNLOCK(&pFlash);
817
818 return status;
819}

866void HAL_FLASH_IRQHandler(void)
867{
868 uint32_t param = 0xFFFFFFFFU;
869 uint32_t error;
870
871 /* Save flash errors. Only ECC detection can be checked here as ECCC
872 generates NMI */
873 error = (FLASH->SR & FLASH_FLAG_SR_ERROR);
874
875 CLEAR_BIT(FLASH->CR, pFlash.ProcedureOnGoing);
876
877 /* A] Set parameter for user or error callbacks */
878 /* check operation was a program or erase */
879 if ((pFlash.ProcedureOnGoing & (FLASH_TYPEPROGRAM_PAGE)) != 0x00U)
880 {
881 /* return adress being programmed */
882 param = pFlash.Address;
883 }else if ((pFlash.ProcedureOnGoing & (FLASH_TYPEERASE_MASSERASE | FLASH_TYPEERASE_SECTORERASE | FLASH_TYPEERASE_PAGEERASE)) != 0x00U)
884 {
885 /* return page number being erased (0 for mass erase) */
886 param = pFlash.PageOrSector;
887 }else
888 {
889 /* Nothing to do */
890 }
891
892 /* B] Check errors */
893 if (error != 0x00U)
894 {
895 /*Save the error code*/
896 pFlash.ErrorCode |= error;
897
898 /* clear error flags */
899 __HAL_FLASH_CLEAR_FLAG(error);
900
901 /*Stop the procedure ongoing*/
902 pFlash.ProcedureOnGoing = FLASH_TYPENONE;
903
904 /* Error callback */
905 HAL_FLASH_OperationErrorCallback(param);
906 }
907
908 /* C] Check FLASH End of Operation flag */
909 if (__HAL_FLASH_GET_FLAG(FLASH_FLAG_EOP) != 0x00U)
910 {
911 /* Clear FLASH End of Operation pending bit */
912 __HAL_FLASH_CLEAR_FLAG(FLASH_FLAG_EOP);
913
914 if (pFlash.ProcedureOnGoing == FLASH_TYPEERASE_PAGEERASE)
915 {
916 /* Nb of pages to erased can be decreased */
917 pFlash.NbPagesSectorsToErase--;
918
919 /* Check if there are still pages to erase*/
920 if (pFlash.NbPagesSectorsToErase != 0x00U)
921 {
922 /* Increment page number */
923 pFlash.PageOrSector += FLASH_PAGE_SIZE;
924 FLASH_PageErase(pFlash.PageOrSector);
925 }else
926 {
927 /* No more pages to erase: stop erase pages procedure */
928 pFlash.ProcedureOnGoing = FLASH_TYPENONE;
929 }
930 }
931 else if (pFlash.ProcedureOnGoing == FLASH_TYPEERASE_SECTORERASE)
932 {
933 /* Nb of sectors to erased can be decreased */
934 pFlash.NbPagesSectorsToErase--;
935
936 /* Check if there are still pages to erase*/
937 if (pFlash.NbPagesSectorsToErase != 0x00U)
938 {
939 /* Increment page number */
940 pFlash.PageOrSector += FLASH_SECTOR_SIZE;
941 FLASH_SectorErase(pFlash.PageOrSector);
942 }else
943 {
944 /* No more pages to erase: stop erase pages procedure */
945 pFlash.ProcedureOnGoing = FLASH_TYPENONE;
946 }
947 }
948 else
949 {
950 /*Stop the ongoing procedure */
951 pFlash.ProcedureOnGoing = FLASH_TYPENONE;
952 }
953
954 /* User callback */
955 HAL_FLASH_EndOfOperationCallback(param);
956 }
