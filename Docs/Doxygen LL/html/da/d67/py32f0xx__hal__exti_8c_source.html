148HAL_StatusTypeDef HAL_EXTI_SetConfigLine(EXTI_HandleTypeDef *hexti, EXTI_ConfigTypeDef *pExtiConfig)
149{
150 uint32_t regval;
151 uint32_t linepos;
152 uint32_t maskline;
153
154 /* Check null pointer */
155 if ((hexti == NULL) || (pExtiConfig == NULL))
156 {
157 return HAL_ERROR;
158 }
159
160 /* Check parameters */
161 assert_param(IS_EXTI_LINE(pExtiConfig->Line));
162 assert_param(IS_EXTI_MODE(pExtiConfig->Mode));
163
164 /* Assign line number to handle */
165 hexti->Line = pExtiConfig->Line;
166
167 /* Compute line mask */
168 linepos = (pExtiConfig->Line & EXTI_PIN_MASK);
169 maskline = (1uL << linepos);
170
171 /* Configure triggers for configurable lines */
172 if ((pExtiConfig->Line & EXTI_CONFIG) != 0x00u)
173 {
174 assert_param(IS_EXTI_TRIGGER(pExtiConfig->Trigger));
175
176 /* Configure rising trigger */
177 /* Mask or set line */
178 if ((pExtiConfig->Trigger & EXTI_TRIGGER_RISING) != 0x00u)
179 {
180 EXTI->RTSR |= maskline;
181 }
182 else
183 {
184 EXTI->RTSR &= ~maskline;
185 }
186
187 /* Configure falling trigger */
188 /* Mask or set line */
189 if ((pExtiConfig->Trigger & EXTI_TRIGGER_FALLING) != 0x00u)
190 {
191 EXTI->FTSR |= maskline;
192 }
193 else
194 {
195 EXTI->FTSR &= ~maskline;
196 }
197
198
199 /* Configure gpio port selection in case of gpio exti line */
200 if ((pExtiConfig->Line & EXTI_GPIO) == EXTI_GPIO)
201 {
202 assert_param(IS_EXTI_GPIO_PORT(pExtiConfig->GPIOSel));
203 assert_param(IS_EXTI_GPIO_PIN(linepos));
204
205 regval = EXTI->EXTICR[linepos >> 2u];
206 regval &= ~(EXTI_EXTICR1_EXTI0 << (EXTI_EXTICR1_EXTI1_Pos * (linepos & 0x03u)));
207 regval |= (pExtiConfig->GPIOSel << (EXTI_EXTICR1_EXTI1_Pos * (linepos & 0x03u)));
208 EXTI->EXTICR[linepos >> 2u] = regval;
209 }
210 }
211
212 /* Configure interrupt mode : read current mode */
213 /* Mask or set line */
214 if ((pExtiConfig->Mode & EXTI_MODE_INTERRUPT) != 0x00u)
215 {
216 EXTI->IMR |= maskline;
217 }
218 else
219 {
220 EXTI->IMR &= ~maskline;
221 }
222
223 /* Configure event mode : read current mode */
224 /* Mask or set line */
225 if ((pExtiConfig->Mode & EXTI_MODE_EVENT) != 0x00u)
226 {
227 EXTI->EMR |= maskline;
228 }
229 else
230 {
231 EXTI->EMR &= ~maskline;
232 }
233
234 return HAL_OK;
235}

243HAL_StatusTypeDef HAL_EXTI_GetConfigLine(EXTI_HandleTypeDef *hexti, EXTI_ConfigTypeDef *pExtiConfig)
244{
245 uint32_t regval;
246 uint32_t linepos;
247 uint32_t maskline;
248
249 /* Check null pointer */
250 if ((hexti == NULL) || (pExtiConfig == NULL))
251 {
252 return HAL_ERROR;
253 }
254
255 /* Check the parameter */
256 assert_param(IS_EXTI_LINE(hexti->Line));
257
258 /* Store handle line number to configuration structure */
259 pExtiConfig->Line = hexti->Line;
260
261 /* Compute line mask */
262 linepos = (pExtiConfig->Line & EXTI_PIN_MASK);
263 maskline = (1uL << linepos);
264
265 /* 1] Get core mode : interrupt */
266
267 /* Check if selected line is enable */
268 if ((EXTI->IMR & maskline) != 0x00u)
269 {
270 pExtiConfig->Mode = EXTI_MODE_INTERRUPT;
271 }
272 else
273 {
274 pExtiConfig->Mode = EXTI_MODE_NONE;
275 }
276
277 /* Get event mode */
278 /* Check if selected line is enable */
279 if ((EXTI->EMR & maskline) != 0x00u)
280 {
281 pExtiConfig->Mode |= EXTI_MODE_EVENT;
282 }
283
284 /* 2] Get trigger for configurable lines : rising */
285 if ((pExtiConfig->Line & EXTI_CONFIG) != 0x00u)
286 {
287 /* Check if configuration of selected line is enable */
288 if ((EXTI->RTSR & maskline) != 0x00u)
289 {
290 pExtiConfig->Trigger = EXTI_TRIGGER_RISING;
291 }
292 else
293 {
294 pExtiConfig->Trigger = EXTI_TRIGGER_NONE;
295 }
296
297 /* Get falling configuration */
298 /* Check if configuration of selected line is enable */
299 if ((EXTI->FTSR & maskline) != 0x00u)
300 {
301 pExtiConfig->Trigger |= EXTI_TRIGGER_FALLING;
302 }
303
304 /* Get Gpio port selection for gpio lines */
305 if ((pExtiConfig->Line & EXTI_GPIO) == EXTI_GPIO)
306 {
307 assert_param(IS_EXTI_GPIO_PIN(linepos));
308
309 regval = EXTI->EXTICR[linepos >> 2u];
310 pExtiConfig->GPIOSel = ((regval << (EXTI_EXTICR1_EXTI1_Pos * (3uL - (linepos & 0x03u)))) >> 24);
311 }
312 else
313 {
314 pExtiConfig->GPIOSel = 0x00u;
315 }
316 }
317 else
318 {
319 /* No Trigger selected */
320 pExtiConfig->Trigger = EXTI_TRIGGER_NONE;
321 pExtiConfig->GPIOSel = 0x00u;
322 }
323
324 return HAL_OK;
325}

332HAL_StatusTypeDef HAL_EXTI_ClearConfigLine(EXTI_HandleTypeDef *hexti)
333{
334 uint32_t regval;
335 uint32_t linepos;
336 uint32_t maskline;
337
338 /* Check null pointer */
339 if (hexti == NULL)
340 {
341 return HAL_ERROR;
342 }
343
344 /* Check the parameter */
345 assert_param(IS_EXTI_LINE(hexti->Line));
346
347 /* compute line mask */
348 linepos = (hexti->Line & EXTI_PIN_MASK);
349 maskline = (1uL << linepos);
350
351 /* 1] Clear interrupt mode */
352 EXTI->IMR = (EXTI->IMR & ~maskline);
353
354 /* 2] Clear event mode */
355 EXTI->EMR = (EXTI->EMR & ~maskline);
356
357 /* 3] Clear triggers in case of configurable lines */
358 if ((hexti->Line & EXTI_CONFIG) != 0x00u)
359 {
360 EXTI->RTSR = (EXTI->RTSR & ~maskline);
361 EXTI->FTSR = (EXTI->FTSR & ~maskline);
362
363 /* Get Gpio port selection for gpio lines */
364 if ((hexti->Line & EXTI_GPIO) == EXTI_GPIO)
365 {
366 assert_param(IS_EXTI_GPIO_PIN(linepos));
367
368 regval = EXTI->EXTICR[linepos >> 2u];
369 regval &= ~(EXTI_EXTICR1_EXTI0 << (EXTI_EXTICR1_EXTI1_Pos * (linepos & 0x03u)));
370 EXTI->EXTICR[linepos >> 2u] = regval;
371 }
372 }
373
374 return HAL_OK;
375}

450void HAL_EXTI_IRQHandler(EXTI_HandleTypeDef *hexti)
451{
452 uint32_t regval;
453 uint32_t maskline;
454
455 /* Compute line mask */
456 maskline = (1uL << (hexti->Line & EXTI_PIN_MASK));
457
458 /* Get pending bit */
459 regval = (EXTI->PR & maskline);
460 if (regval != 0x00u)
461 {
462 /* Clear pending bit */
463 EXTI->PR = maskline;
464
465 /* Call callback */
466 if (hexti->PendingCallback != NULL)
467 {
468 hexti->PendingCallback();
