741{
742 uint16_t *tmp;
743 uint32_t tickstart = 0U;
744
745 if (husart->State == HAL_USART_STATE_READY)
746 {
747 if ((pTxData == NULL) || (Size == 0))
748 {
749 return HAL_ERROR;
750 }
751
752 /* Process Locked */
753 __HAL_LOCK(husart);
754
755 husart->ErrorCode = HAL_USART_ERROR_NONE;
756 husart->State = HAL_USART_STATE_BUSY_TX;
757
758 /* Init tickstart for timeout management */
759 tickstart = HAL_GetTick();
760
761 husart->TxXferSize = Size;
762 husart->TxXferCount = Size;
763 while (husart->TxXferCount > 0U)
764 {
765 husart->TxXferCount--;
766 if (husart->Init.WordLength == USART_WORDLENGTH_9B)
767 {
768 /* Wait for TC flag in order to write data in DR */
769 if (USART_WaitOnFlagUntilTimeout(husart, USART_FLAG_TXE, RESET, tickstart, Timeout) != HAL_OK)
770 {
771 return HAL_TIMEOUT;
772 }
773 tmp = (uint16_t *) pTxData;
774 husart->Instance->DR = (*tmp & (uint16_t)0x01FF);
775 if (husart->Init.Parity == USART_PARITY_NONE)
776 {
777 pTxData += 2U;
778 }
779 else
780 {
781 pTxData += 1U;
782 }
783 }
784 else
785 {
786 if (USART_WaitOnFlagUntilTimeout(husart, USART_FLAG_TXE, RESET, tickstart, Timeout) != HAL_OK)

788 return HAL_TIMEOUT;
789 }
790 husart->Instance->DR = (*pTxData++ & (uint8_t)0xFF);
791 }
792 }

826{

864 return HAL_TIMEOUT;

876 }
877 }

1289 return HAL_ERROR;

1348 in the USART CR3 register */

1571HAL_StatusTypeDef HAL_USART_Abort(USART_HandleTypeDef *husart)
1572{
1573 /* Disable TXEIE, TCIE, RXNE, PE and ERR (Frame error, noise error, overrun error) interrupts */
1574 CLEAR_BIT(husart->Instance->CR1, (USART_CR1_RXNEIE | USART_CR1_PEIE | USART_CR1_TXEIE | USART_CR1_TCIE));
1575 CLEAR_BIT(husart->Instance->CR3, USART_CR3_EIE);
1576
1577#if (defined(DMA1) || defined(DMA))
1578 /* Disable the USART DMA Tx request if enabled */
1579 if (HAL_IS_BIT_SET(husart->Instance->CR3, USART_CR3_DMAT))
1580 {
1581 CLEAR_BIT(husart->Instance->CR3, USART_CR3_DMAT);
1582
1583 /* Abort the USART DMA Tx channel : use blocking DMA Abort API (no callback) */
1584 if (husart->hdmatx != NULL)
1585 {
1586 /* Set the USART DMA Abort callback to Null.
1587 No call back execution at end of DMA abort procedure */
1588 husart->hdmatx->XferAbortCallback = NULL;
1589
1590 HAL_DMA_Abort(husart->hdmatx);
1591 }
1592 }
1593
1594 /* Disable the USART DMA Rx request if enabled */
1595 if (HAL_IS_BIT_SET(husart->Instance->CR3, USART_CR3_DMAR))
1596 {
1597 CLEAR_BIT(husart->Instance->CR3, USART_CR3_DMAR);
1598
1599 /* Abort the USART DMA Rx channel : use blocking DMA Abort API (no callback) */
1600 if (husart->hdmarx != NULL)
1601 {
1602 /* Set the USART DMA Abort callback to Null.
1603 No call back execution at end of DMA abort procedure */
1604 husart->hdmarx->XferAbortCallback = NULL;
1605
1606 HAL_DMA_Abort(husart->hdmarx);
1607 }
1608 }
1609#endif /* DMA1 or DMA */
1610
1611 /* Reset Tx and Rx transfer counters */
1612 husart->TxXferCount = 0x00U;
1613 husart->RxXferCount = 0x00U;
1614
1615 /* Restore husart->State to Ready */
1616 husart->State = HAL_USART_STATE_READY;
1617
1618 /* Reset Handle ErrorCode to No Error */
1619 husart->ErrorCode = HAL_USART_ERROR_NONE;
1620
1621 return HAL_OK;

1641HAL_StatusTypeDef HAL_USART_Abort_IT(USART_HandleTypeDef *husart)
1642{
1643 uint32_t AbortCplt = 0x01U;
1644
1645 /* Disable TXEIE, TCIE, RXNE, PE and ERR (Frame error, noise error, overrun error) interrupts */
1646 CLEAR_BIT(husart->Instance->CR1, (USART_CR1_RXNEIE | USART_CR1_PEIE | USART_CR1_TXEIE | USART_CR1_TCIE));
1647 CLEAR_BIT(husart->Instance->CR3, USART_CR3_EIE);
1648
1649#if (defined(DMA1) || defined(DMA))
1650 /* If DMA Tx and/or DMA Rx Handles are associated to USART Handle, DMA Abort complete callbacks should be initialised
1651 before any call to DMA Abort functions */
1652 /* DMA Tx Handle is valid */
1653 if (husart->hdmatx != NULL)
1654 {
1655 /* Set DMA Abort Complete callback if USART DMA Tx request if enabled.
1656 Otherwise, set it to NULL */
1657 if (HAL_IS_BIT_SET(husart->Instance->CR3, USART_CR3_DMAT))
1658 {
1659 husart->hdmatx->XferAbortCallback = USART_DMATxAbortCallback;
1660 }
1661 else
1662 {
1663 husart->hdmatx->XferAbortCallback = NULL;
1664 }
1665 }
1666 /* DMA Rx Handle is valid */
1667 if (husart->hdmarx != NULL)
1668 {
1669 /* Set DMA Abort Complete callback if USART DMA Rx request if enabled.
1670 Otherwise, set it to NULL */
1671 if (HAL_IS_BIT_SET(husart->Instance->CR3, USART_CR3_DMAR))
1672 {
1673 husart->hdmarx->XferAbortCallback = USART_DMARxAbortCallback;
1674 }
1675 else
1676 {
1677 husart->hdmarx->XferAbortCallback = NULL;
1678 }
1679 }
1680
1681 /* Disable the USART DMA Tx request if enabled */
1682 if (HAL_IS_BIT_SET(husart->Instance->CR3, USART_CR3_DMAT))
1683 {
1684 /* Disable DMA Tx at USART level */
1685 CLEAR_BIT(husart->Instance->CR3, USART_CR3_DMAT);
1686
1687 /* Abort the USART DMA Tx channel : use non blocking DMA Abort API (callback) */
1688 if (husart->hdmatx != NULL)
1689 {
1690 /* USART Tx DMA Abort callback has already been initialised :
1691 will lead to call HAL_USART_AbortCpltCallback() at end of DMA abort procedure */
1692
1693 /* Abort DMA TX */
1694 if (HAL_DMA_Abort_IT(husart->hdmatx) != HAL_OK)
1695 {
1696 husart->hdmatx->XferAbortCallback = NULL;
1697 }
1698 else
1699 {
1700 AbortCplt = 0x00U;
1701 }
1702 }
1703 }
1704
1705 /* Disable the USART DMA Rx request if enabled */
1706 if (HAL_IS_BIT_SET(husart->Instance->CR3, USART_CR3_DMAR))
1707 {
1708 CLEAR_BIT(husart->Instance->CR3, USART_CR3_DMAR);
1709
1710 /* Abort the USART DMA Rx channel : use non blocking DMA Abort API (callback) */
1711 if (husart->hdmarx != NULL)
1712 {
1713 /* USART Rx DMA Abort callback has already been initialised :
1714 will lead to call HAL_USART_AbortCpltCallback() at end of DMA abort procedure */
1715
1716 /* Abort DMA RX */
1717 if (HAL_DMA_Abort_IT(husart->hdmarx) != HAL_OK)
1718 {
1719 husart->hdmarx->XferAbortCallback = NULL;
1720 AbortCplt = 0x01U;
1721 }
1722 else
1723 {
1724 AbortCplt = 0x00U;
1725 }
1726 }
1727 }
1728#endif /* DMA1 or DMA */
1729
1730 /* if no DMA abort complete callback execution is required => call user Abort Complete callback */
1731 if (AbortCplt == 0x01U)
1732 {
1733 /* Reset Tx and Rx transfer counters */
1734 husart->TxXferCount = 0x00U;
1735 husart->RxXferCount = 0x00U;
1736
1737 /* Reset errorCode */
1738 husart->ErrorCode = HAL_USART_ERROR_NONE;
1739
1740 /* Restore husart->State to Ready */
1741 husart->State = HAL_USART_STATE_READY;
1742
1743 /* As no DMA to be aborted, call directly user Abort complete callback */
1744#if (USE_HAL_USART_REGISTER_CALLBACKS == 1)
1745 /* Call registered Abort Complete Callback */
1746 husart->AbortCpltCallback(husart);
1747#else
1748 /* Call legacy weak Abort Complete Callback */
1749 HAL_USART_AbortCpltCallback(husart);
1750#endif /* USE_HAL_USART_REGISTER_CALLBACKS */
1751 }
1752
1753 return HAL_OK;
1754}

1764void HAL_USART_IRQHandler(USART_HandleTypeDef *husart)
1765{
1766 uint32_t isrflags = READ_REG(husart->Instance->SR);
1767 uint32_t cr1its = READ_REG(husart->Instance->CR1);
1768 uint32_t cr3its = READ_REG(husart->Instance->CR3);
1769 uint32_t errorflags = 0x00U;
1770#if (defined(DMA1) || defined(DMA))
1771 uint32_t dmarequest = 0x00U;
1772#endif /* DMA1 or DMA */
1773
1774 /* If no error occurs */
1775 errorflags = (isrflags & (uint32_t)(USART_SR_PE | USART_SR_FE | USART_SR_ORE | USART_SR_NE));
1776 if (errorflags == RESET)
1777 {
1778 /* USART in mode Receiver -------------------------------------------------*/
1779 if (((isrflags & USART_SR_RXNE) != RESET) && ((cr1its & USART_CR1_RXNEIE) != RESET))
1780 {
1781 if (husart->State == HAL_USART_STATE_BUSY_RX)
1782 {
1783 USART_Receive_IT(husart);
1784 }
1785 else
1786 {
1787 USART_TransmitReceive_IT(husart);
1788 }
1789 return;
1790 }
1791 }
1792 /* If some errors occur */
1793 if ((errorflags != RESET) && (((cr3its & USART_CR3_EIE) != RESET) || ((cr1its & (USART_CR1_RXNEIE | USART_CR1_PEIE)) != RESET)))
1794 {
1795 /* USART parity error interrupt occurred ----------------------------------*/
1796 if (((isrflags & USART_SR_PE) != RESET) && ((cr1its & USART_CR1_PEIE) != RESET))
1797 {
1798 husart->ErrorCode |= HAL_USART_ERROR_PE;
1799 }
1800
1801 /* USART noise error interrupt occurred --------------------------------*/
1802 if (((isrflags & USART_SR_NE) != RESET) && ((cr3its & USART_CR3_EIE) != RESET))
1803 {
1804 husart->ErrorCode |= HAL_USART_ERROR_NE;
1805 }
1806
1807 /* USART frame error interrupt occurred --------------------------------*/
1808 if (((isrflags & USART_SR_FE) != RESET) && ((cr3its & USART_CR3_EIE) != RESET))
1809 {
1810 husart->ErrorCode |= HAL_USART_ERROR_FE;
1811 }
1812
1813 /* USART Over-Run interrupt occurred -----------------------------------*/
1814 if (((isrflags & USART_SR_ORE) != RESET) && ((cr3its & USART_CR3_EIE) != RESET))
1815 {
1816 husart->ErrorCode |= HAL_USART_ERROR_ORE;
1817 }
1818
1819 if (husart->ErrorCode != HAL_USART_ERROR_NONE)
1820 {
1821 /* USART in mode Receiver -----------------------------------------------*/
1822 if (((isrflags & USART_SR_RXNE) != RESET) && ((cr1its & USART_CR1_RXNEIE) != RESET))
1823 {
1824 if (husart->State == HAL_USART_STATE_BUSY_RX)
1825 {
1826 USART_Receive_IT(husart);
1827 }
1828 else
1829 {
1830 USART_TransmitReceive_IT(husart);
1831 }
1832 }
1833 /* If Overrun error occurs, or if any error occurs in DMA mode reception,
1834 consider error as blocking */
1835#if (defined(DMA1) || defined(DMA))
1836 dmarequest = HAL_IS_BIT_SET(husart->Instance->CR3, USART_CR3_DMAR);
1837 if (((husart->ErrorCode & HAL_USART_ERROR_ORE) != RESET) || dmarequest)
1838#else
1839 if ((husart->ErrorCode & HAL_USART_ERROR_ORE) != RESET)
1840#endif /* DMA1 or DMA */
1841 {
1842 /* Set the USART state ready to be able to start again the process,
1843 Disable Rx Interrupts, and disable Rx DMA request, if ongoing */
1844 USART_EndRxTransfer(husart);
1845
1846#if (defined(DMA1) || defined(DMA))
1847 /* Disable the USART DMA Rx request if enabled */
1848 if (HAL_IS_BIT_SET(husart->Instance->CR3, USART_CR3_DMAR))
1849 {
1850 CLEAR_BIT(husart->Instance->CR3, USART_CR3_DMAR);
1851
1852 /* Abort the USART DMA Rx channel */
1853 if (husart->hdmarx != NULL)
1854 {
1855 /* Set the USART DMA Abort callback :
1856 will lead to call HAL_USART_ErrorCallback() at end of DMA abort procedure */
1857 husart->hdmarx->XferAbortCallback = USART_DMAAbortOnError;
1858
1859 if (HAL_DMA_Abort_IT(husart->hdmarx) != HAL_OK)
1860 {
1861 /* Call Directly XferAbortCallback function in case of error */
1862 husart->hdmarx->XferAbortCallback(husart->hdmarx);
1863 }
1864 }
1865 else
1866 {
1867 /* Call user error callback */
1868#if (USE_HAL_USART_REGISTER_CALLBACKS == 1)
1869 /* Call registered Error Callback */
1870 husart->ErrorCallback(husart);
1871#else
1872 /* Call legacy weak Error Callback */
1873 HAL_USART_ErrorCallback(husart);
1874#endif /* USE_HAL_USART_REGISTER_CALLBACKS */
1875 }
1876 }
1877 else
1878 {
1879#endif /* DMA1 or DMA */
1880 /* Call user error callback */
1881#if (USE_HAL_USART_REGISTER_CALLBACKS == 1)
1882 /* Call registered Error Callback */
1883 husart->ErrorCallback(husart);
1884#else
1885 /* Call legacy weak Error Callback */
1886 HAL_USART_ErrorCallback(husart);
1887#endif /* USE_HAL_USART_REGISTER_CALLBACKS */
1888#if (defined(DMA1) || defined(DMA))
1889 }
1890#endif /* DMA1 or DMA */
1891 }
1892 else
1893 {
1894 /* Call user error callback */
1895#if (USE_HAL_USART_REGISTER_CALLBACKS == 1)
1896 /* Call registered Error Callback */
1897 husart->ErrorCallback(husart);
1898#else
1899 /* Call legacy weak Error Callback */
1900 HAL_USART_ErrorCallback(husart);
1901#endif /* USE_HAL_USART_REGISTER_CALLBACKS */
1902 husart->ErrorCode = HAL_USART_ERROR_NONE;
1903 }
1904 }
1905 return;
1906 }
1907
1908 /* USART in mode Transmitter -----------------------------------------------*/
1909 if (((isrflags & USART_SR_TXE) != RESET) && ((cr1its & USART_CR1_TXEIE) != RESET))
1910 {
1911 if (husart->State == HAL_USART_STATE_BUSY_TX)
1912 {
1913 USART_Transmit_IT(husart);
1914 }
1915 else
1916 {
1917 USART_TransmitReceive_IT(husart);
1918 }
