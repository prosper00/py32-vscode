203HAL_StatusTypeDef HAL_RCC_DeInit(void)
204{
205 uint32_t tickstart;
206
207 /* Get Start Tick*/
208 tickstart = HAL_GetTick();
209
210 /* Set HSION bit to the reset value */
211 SET_BIT(RCC->CR, RCC_CR_HSION);
212
213 /* Wait till HSI is ready */
214 while (READ_BIT(RCC->CR, RCC_CR_HSIRDY) == 0U)
215 {
216 if ((HAL_GetTick() - tickstart) > HSI_TIMEOUT_VALUE)
217 {
218 return HAL_TIMEOUT;
219 }
220 }
221
222 /* Set ICSCR to the reset value */
223 RCC->ICSCR = 0x00FF10FF;
224
225 /* Get Start Tick*/
226 tickstart = HAL_GetTick();
227
228 /* Reset CFGR register (HSI is selected as system clock source) */
229 RCC->CFGR = 0x00000000u;
230
231 /* Wait till HSI is ready */
232 while (READ_BIT(RCC->CFGR, RCC_CFGR_SWS) != 0U)
233 {
234 if ((HAL_GetTick() - tickstart) > CLOCKSWITCH_TIMEOUT_VALUE)
235 {
236 return HAL_TIMEOUT;
237 }
238 }
239
240 /* Clear CR register in 2 steps: first to clear HSEON in case bypass was enabled */
241 RCC->CR = RCC_CR_HSION;
242
243 /* Then again to HSEBYP in case bypass was enabled */
244 RCC->CR = RCC_CR_HSION;
245
246#if defined(RCC_PLL_SUPPORT)
247 /* Get Start Tick*/
248 tickstart = HAL_GetTick();
249
250 /* Wait till PLL is ready */
251 while (READ_BIT(RCC->CR, RCC_CR_PLLRDY) != 0U)
252 {
253 if ((HAL_GetTick() - tickstart) > PLL_TIMEOUT_VALUE)
254 {
255 return HAL_TIMEOUT;
256 }
257 }
258
259 /* once PLL is OFF, reset PLLCFGR register to default value */
260 RCC->PLLCFGR = 0x00000000u;
261#endif
262 /* Disable all interrupts */
263 RCC->CIER = 0x00000000u;
264
265 /* Clear all flags */
266 RCC->CICR = 0xFFFFFFFFu;
267
268 /* Update the SystemCoreClock global variable */
269 SystemCoreClock = HSI_VALUE;
270
271 /* Adapt Systick interrupt period */
272 if (HAL_InitTick(uwTickPrio) != HAL_OK)
273 {
274 return HAL_ERROR;
275 }
276 else
277 {
278 return HAL_OK;
279 }

778HAL_StatusTypeDef HAL_RCC_ClockConfig(RCC_ClkInitTypeDef *RCC_ClkInitStruct, uint32_t FLatency)
779{
780 uint32_t tickstart;
781
782 /* Check Null pointer */
783 if (RCC_ClkInitStruct == NULL)
784 {
785 return HAL_ERROR;
786 }
787
788 /* Check the parameters */
789 assert_param(IS_RCC_CLOCKTYPE(RCC_ClkInitStruct->ClockType));
790 assert_param(IS_FLASH_LATENCY(FLatency));
791
792 /* To correctly read data from FLASH memory, the number of wait states (LATENCY)
793 must be correctly programmed according to the frequency of the FLASH clock
794 (HCLK) and the supply voltage of the device. */
795
796 /* Increasing the number of wait states because of higher CPU frequency */
797 if (FLatency > __HAL_FLASH_GET_LATENCY())
798 {
799 /* Program the new number of wait states to the LATENCY bits in the FLASH_ACR register */
800 __HAL_FLASH_SET_LATENCY(FLatency);
801
802 /* Check that the new number of wait states is taken into account to access the Flash
803 memory by polling the FLASH_ACR register */
804 tickstart = HAL_GetTick();
805
806 while ((FLASH->ACR & FLASH_ACR_LATENCY) != FLatency)
807 {
808 if ((HAL_GetTick() - tickstart) > CLOCKSWITCH_TIMEOUT_VALUE)
809 {
810 return HAL_TIMEOUT;
811 }
812 }
813 }
814
815 /*-------------------------- HCLK Configuration --------------------------*/
816 if (((RCC_ClkInitStruct->ClockType) & RCC_CLOCKTYPE_HCLK) == RCC_CLOCKTYPE_HCLK)
817 {
818 /* Set the highest APB divider in order to ensure that we do not go through
819 a non-spec phase whatever we decrease or increase HCLK. */
820 if (((RCC_ClkInitStruct->ClockType) & RCC_CLOCKTYPE_PCLK1) == RCC_CLOCKTYPE_PCLK1)
821 {
822 MODIFY_REG(RCC->CFGR, RCC_CFGR_PPRE, RCC_HCLK_DIV16);
823 }
824
825 /* Set the new HCLK clock divider */
826 assert_param(IS_RCC_HCLK(RCC_ClkInitStruct->AHBCLKDivider));
827 MODIFY_REG(RCC->CFGR, RCC_CFGR_HPRE, RCC_ClkInitStruct->AHBCLKDivider);
828 }
829
830 /*------------------------- SYSCLK Configuration ---------------------------*/
831 if (((RCC_ClkInitStruct->ClockType) & RCC_CLOCKTYPE_SYSCLK) == RCC_CLOCKTYPE_SYSCLK)
832 {
833 assert_param(IS_RCC_SYSCLKSOURCE(RCC_ClkInitStruct->SYSCLKSource));
834
835 /* HSE is selected as System Clock Source */
836 if (RCC_ClkInitStruct->SYSCLKSource == RCC_SYSCLKSOURCE_HSE)
837 {
838 /* Check the HSE ready flag */
839 if (READ_BIT(RCC->CR, RCC_CR_HSERDY) == 0U)
840 {
841 return HAL_ERROR;
842 }
843 }
844 /* HSI is selected as System Clock Source */
845 else if (RCC_ClkInitStruct->SYSCLKSource == RCC_SYSCLKSOURCE_HSI)
846 {
847 /* Check the HSI ready flag */
848 if (READ_BIT(RCC->CR, RCC_CR_HSIRDY) == 0U)
849 {
850 return HAL_ERROR;
851 }
852 }
853#if defined(RCC_PLL_SUPPORT)
854 /* PLL is selected as System Clock Source */
855 else if (RCC_ClkInitStruct->SYSCLKSource == RCC_SYSCLKSOURCE_PLLCLK)
856 {
857 /* Check the PLL ready flag */
858 if (READ_BIT(RCC->CR, RCC_CR_PLLRDY) == 0U)
859 {
860 return HAL_ERROR;
861 }
862 }
863#endif
864#if defined(RCC_LSE_SUPPORT)
865 /* LSE is selected as System Clock Source */
866 else if (RCC_ClkInitStruct->SYSCLKSource == RCC_SYSCLKSOURCE_LSE)
867 {
868 /* Check the LSE ready flag */
869 if (READ_BIT(RCC->BDCR, RCC_BDCR_LSERDY) == 0U)
870 {
871 return HAL_ERROR;
872 }
873 }
874#endif
875 /* LSI is selected as System Clock Source */
876 else
877 {
878 /* Check the LSI ready flag */
879 if (READ_BIT(RCC->CSR, RCC_CSR_LSIRDY) == 0U)
880 {
881 return HAL_ERROR;
882 }
883 }
884 MODIFY_REG(RCC->CFGR, RCC_CFGR_SW, RCC_ClkInitStruct->SYSCLKSource);
885
886 /* Get Start Tick*/
887 tickstart = HAL_GetTick();
888
889 while (__HAL_RCC_GET_SYSCLK_SOURCE() != (RCC_ClkInitStruct->SYSCLKSource << RCC_CFGR_SWS_Pos))
890 {
891 if ((HAL_GetTick() - tickstart) > CLOCKSWITCH_TIMEOUT_VALUE)
892 {
893 return HAL_TIMEOUT;
894 }
895 }
896 }
897
898 /* Decreasing the number of wait states because of lower CPU frequency */
899 if (FLatency < __HAL_FLASH_GET_LATENCY())
900 {
901 /* Program the new number of wait states to the LATENCY bits in the FLASH_ACR register */
902 __HAL_FLASH_SET_LATENCY(FLatency);
903
904 /* Check that the new number of wait states is taken into account to access the Flash
905 memory by polling the FLASH_ACR register */
906 tickstart = HAL_GetTick();
907
908 while ((FLASH->ACR & FLASH_ACR_LATENCY) != FLatency)
909 {
910 if ((HAL_GetTick() - tickstart) > CLOCKSWITCH_TIMEOUT_VALUE)
911 {
912 return HAL_TIMEOUT;
913 }
914 }
915 }
916
917 /*-------------------------- PCLK1 Configuration ---------------------------*/
918 if (((RCC_ClkInitStruct->ClockType) & RCC_CLOCKTYPE_PCLK1) == RCC_CLOCKTYPE_PCLK1)
919 {
920 assert_param(IS_RCC_PCLK(RCC_ClkInitStruct->APB1CLKDivider));
921 MODIFY_REG(RCC->CFGR, RCC_CFGR_PPRE, RCC_ClkInitStruct->APB1CLKDivider);
922 }
923
924 /* Update the SystemCoreClock global variable */
925 SystemCoreClock = (HAL_RCC_GetSysClockFreq() >> ((AHBPrescTable[(RCC->CFGR & RCC_CFGR_HPRE) >> RCC_CFGR_HPRE_Pos]) & 0x1FU));
926
927 /* Configure the source of time base considering new system clocks settings*/
928 return HAL_InitTick(uwTickPrio);
929}

1089uint32_t HAL_RCC_GetSysClockFreq(void)
1090{
1091 uint32_t hsidiv;
1092 uint32_t sysclockfreq;
1093 const uint32_t hsiValue[5] = {4000000,8000000,16000000,22120000,24000000};
1094 uint32_t hsiIndex;
1095#if defined(RCC_PLL_SUPPORT)
1096 uint32_t pllsource;
1097#endif
1098
1099 if (__HAL_RCC_GET_SYSCLK_SOURCE() == RCC_CFGR_SWS_HSI)
1100 {
1101 /* HSISYS can be derived for HSI */
1102 hsidiv = (1UL << ((READ_BIT(RCC->CR, RCC_CR_HSIDIV)) >> RCC_CR_HSIDIV_Pos));
1103
1104 /* HSISYS used as system clock source */
1105 hsiIndex = (RCC->ICSCR&RCC_ICSCR_HSI_FS_Msk)>>RCC_ICSCR_HSI_FS_Pos;
1106 if (hsiIndex > 4)
1107 {
1108 hsiIndex = 0;
1109 }
1110 sysclockfreq = (hsiValue[hsiIndex] / hsidiv);
1111 }
1112 else if (__HAL_RCC_GET_SYSCLK_SOURCE() == RCC_CFGR_SWS_HSE)
1113 {
1114 /* HSE used as system clock source */
1115 sysclockfreq = HSE_VALUE;
1116 }
1117#if defined(RCC_PLL_SUPPORT)
1118 else if (__HAL_RCC_GET_SYSCLK_SOURCE() == RCC_CFGR_SWS_PLL)
1119 {
1120 pllsource = (RCC->PLLCFGR & RCC_PLLCFGR_PLLSRC);
1121
1122 switch (pllsource)
1123 {
1124 case RCC_PLLSOURCE_HSE: /* HSE used as PLL clock source */
1125 sysclockfreq = HSE_VALUE * 2;
1126 break;
1127
1128 case RCC_PLLSOURCE_HSI: /* HSI used as PLL clock source */
1129 hsiIndex = (RCC->ICSCR&RCC_ICSCR_HSI_FS_Msk)>>RCC_ICSCR_HSI_FS_Pos;
1130 if (hsiIndex > 4)
1131 {
1132 hsiIndex = 0;
1133 }
1134 sysclockfreq = hsiValue[hsiIndex] * 2;
1135 break;
1136
1137 default: /* HSI used as PLL clock source */
1138 sysclockfreq = hsiValue[4] * 2;
1139 break;
1140 }
1141 }
1142#endif
1143#if defined(RCC_LSE_SUPPORT)
1144 else if (__HAL_RCC_GET_SYSCLK_SOURCE() == RCC_CFGR_SWS_LSE)
1145 {
1146 /* LSE used as system clock source */
1147 sysclockfreq = LSE_VALUE;
1148 }
1149#endif
1150 else if (__HAL_RCC_GET_SYSCLK_SOURCE() == RCC_CFGR_SWS_LSI)
1151 {
1152 /* LSI used as system clock source */
1153 sysclockfreq = LSI_VALUE;
1154 }
1155 else
1156 {
1157 sysclockfreq = 0U;
1158 }
1159
1160 return sysclockfreq;
1161}
