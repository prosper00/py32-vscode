260HAL_StatusTypeDef HAL_RTC_Init(RTC_HandleTypeDef *hrtc)
261{
262 uint32_t prescaler = 0U;
263 /* Check input parameters */
264 if (hrtc == NULL)
265 {
266 return HAL_ERROR;
267 }
268
269 /* Check the parameters */
270 assert_param(IS_RTC_ALL_INSTANCE(hrtc->Instance));
271 assert_param(IS_RTC_CALIB_OUTPUT(hrtc->Init.OutPut));
272 assert_param(IS_RTC_ASYNCH_PREDIV(hrtc->Init.AsynchPrediv));
273
274#if (USE_HAL_RTC_REGISTER_CALLBACKS == 1)
275 if (hrtc->State == HAL_RTC_STATE_RESET)
276 {
277 /* Allocate lock resource and initialize it */
278 hrtc->Lock = HAL_UNLOCKED;
279
280 hrtc->AlarmAEventCallback = HAL_RTC_AlarmAEventCallback; /* Legacy weak AlarmAEventCallback */
281
282 if (hrtc->MspInitCallback == NULL)
283 {
284 hrtc->MspInitCallback = HAL_RTC_MspInit;
285 }
286 /* Init the low level hardware */
287 hrtc->MspInitCallback(hrtc);
288
289 if (hrtc->MspDeInitCallback == NULL)
290 {
291 hrtc->MspDeInitCallback = HAL_RTC_MspDeInit;
292 }
293 }
294#else
295 if (hrtc->State == HAL_RTC_STATE_RESET)
296 {
297 /* Allocate lock resource and initialize it */
298 hrtc->Lock = HAL_UNLOCKED;
299
300 /* Initialize RTC MSP */
301 HAL_RTC_MspInit(hrtc);
302 }
303#endif /* (USE_HAL_RTC_REGISTER_CALLBACKS) */
304
305 /* Set RTC state */
306 hrtc->State = HAL_RTC_STATE_BUSY;
307
308 /* Waiting for synchro */
309 if (HAL_RTC_WaitForSynchro(hrtc) != HAL_OK)
310 {
311 /* Set RTC state */
312 hrtc->State = HAL_RTC_STATE_ERROR;
313
314 return HAL_ERROR;
315 }
316
317 /* Set Initialization mode */
318 if (RTC_EnterInitMode(hrtc) != HAL_OK)
319 {
320 /* Set RTC state */
321 hrtc->State = HAL_RTC_STATE_ERROR;
322
323 return HAL_ERROR;
324 }
325 else
326 {
327 /* Clear Flags Bits */
328 CLEAR_BIT(hrtc->Instance->CRL, (RTC_FLAG_OW | RTC_FLAG_ALRAF | RTC_FLAG_SEC));
329
330 /* Set the signal which will be routed to RTC Tamper pin*/
331 MODIFY_REG(hrtc->Instance->BKP_RTCCR, (BKP_RTCCR_CCO | BKP_RTCCR_ASOE | BKP_RTCCR_ASOS), hrtc->Init.OutPut);
332
333 if (hrtc->Init.AsynchPrediv != RTC_AUTO_1_SECOND)
334 {
335 /* RTC Prescaler provided directly by end-user*/
336 prescaler = hrtc->Init.AsynchPrediv;
337 }
338 else
339 {
340 /* RTC Prescaler will be automatically calculated to get 1 second timebase */
341 /* Get the RTCCLK frequency */
342 prescaler = HAL_RCCEx_GetPeriphCLKFreq(RCC_PERIPHCLK_RTC);
343
344 /* Check that RTC clock is enabled*/
345 if (prescaler == 0U)
346 {
347 /* Should not happen. Frequency is not available*/
348 hrtc->State = HAL_RTC_STATE_ERROR;
349 return HAL_ERROR;
350 }
351 else
352 {
353 /* RTC period = RTCCLK/(RTC_PR + 1) */
354 prescaler = prescaler - 1U;
355 }
356 }
357
358 /* Configure the RTC_PRLH / RTC_PRLL */
359 MODIFY_REG(hrtc->Instance->PRLH, RTC_PRLH_PRL, (prescaler >> 16U));
360 MODIFY_REG(hrtc->Instance->PRLL, RTC_PRLL_PRL, (prescaler & RTC_PRLL_PRL));
361
362 /* Wait for synchro */
363 if (RTC_ExitInitMode(hrtc) != HAL_OK)
364 {
365 hrtc->State = HAL_RTC_STATE_ERROR;
366
367 return HAL_ERROR;
368 }
369
370 /* Initialize date to 1st of January 2000 */
371 hrtc->DateToUpdate.Year = 0x00U;
372 hrtc->DateToUpdate.Month = RTC_MONTH_JANUARY;
373 hrtc->DateToUpdate.Date = 0x01U;
374
375 /* Set RTC state */
376 hrtc->State = HAL_RTC_STATE_READY;
377
378 return HAL_OK;
379 }

406 /* Set RTC state */
407 hrtc->State = HAL_RTC_STATE_ERROR;
408
409 /* Release Lock */

415 {
416 CLEAR_REG(hrtc->Instance->CNTL);
417 CLEAR_REG(hrtc->Instance->CNTH);
418 WRITE_REG(hrtc->Instance->PRLL, 0x00008000U);

420

423 CLEAR_REG(hrtc->Instance->CRL);
424

426 {
427 hrtc->State = HAL_RTC_STATE_ERROR;
428
429 /* Process Unlocked */
430 __HAL_UNLOCK(hrtc);
431
432 return HAL_ERROR;
433 }
434 }
435
436 /* Wait for synchro*/
437 HAL_RTC_WaitForSynchro(hrtc);
438
439 /* Clear RSF flag */
440 CLEAR_BIT(hrtc->Instance->CRL, RTC_FLAG_RSF);
441
442#if (USE_HAL_RTC_REGISTER_CALLBACKS == 1)
443 if (hrtc->MspDeInitCallback == NULL)
444 {
445 hrtc->MspDeInitCallback = HAL_RTC_MspDeInit;
446 }
447
448 /* DeInit the low level hardware: CLOCK, NVIC.*/
449 hrtc->MspDeInitCallback(hrtc);
450
451#else
452 /* De-Initialize RTC MSP */
453 HAL_RTC_MspDeInit(hrtc);
454#endif /* (USE_HAL_RTC_REGISTER_CALLBACKS) */
455
456 hrtc->State = HAL_RTC_STATE_RESET;
457
458 /* Release Lock */
459 __HAL_UNLOCK(hrtc);
460
461 return HAL_OK;
462}

748

776HAL_StatusTypeDef HAL_RTC_GetTime(RTC_HandleTypeDef *hrtc, RTC_TimeTypeDef *sTime, uint32_t Format)
777{
778 uint32_t counter_time = 0U, counter_alarm = 0U, days_elapsed = 0U, hours = 0U;
779
780 /* Check input parameters */
781 if ((hrtc == NULL) || (sTime == NULL))
782 {
783 return HAL_ERROR;
784 }
785
786 /* Check the parameters */
787 assert_param(IS_RTC_FORMAT(Format));
788
789 /* Check if counter overflow occurred */
790 if (__HAL_RTC_OVERFLOW_GET_FLAG(hrtc, RTC_FLAG_OW))
791 {
792 return HAL_ERROR;
793 }
794
795 /* Read the time counter*/
796 counter_time = RTC_ReadTimeCounter(hrtc);
797
798 /* Fill the structure fields with the read parameters */
799 hours = counter_time / 3600U;
800 sTime->Minutes = (uint8_t)((counter_time % 3600U) / 60U);
801 sTime->Seconds = (uint8_t)((counter_time % 3600U) % 60U);
802
803 if (hours >= 24U)
804 {
805 /* Get number of days elapsed from last calculation */
806 days_elapsed = (hours / 24U);
807
808 /* Set Hours in RTC_TimeTypeDef structure*/
809 sTime->Hours = (hours % 24U);
810
811 /* Read Alarm counter in RTC registers */
812 counter_alarm = RTC_ReadAlarmCounter(hrtc);
813
814 /* Calculate remaining time to reach alarm (only if set and not yet expired)*/
815 if ((counter_alarm != RTC_ALARM_RESETVALUE) && (counter_alarm > counter_time))
816 {
817 counter_alarm -= counter_time;
818 }
819 else
820 {
821 /* In case of counter_alarm < counter_time */
822 /* Alarm expiration already occurred but alarm not deactivated */
823 counter_alarm = RTC_ALARM_RESETVALUE;
824 }
825
826 /* Set updated time in decreasing counter by number of days elapsed */
827 counter_time -= (days_elapsed * 24U * 3600U);
828
829 /* Write time counter in RTC registers */
830 if (RTC_WriteTimeCounter(hrtc, counter_time) != HAL_OK)
831 {
832 return HAL_ERROR;
833 }
834
835 /* Set updated alarm to be set */
836 if (counter_alarm != RTC_ALARM_RESETVALUE)
837 {
838 counter_alarm += counter_time;
839
840 /* Write time counter in RTC registers */
841 if (RTC_WriteAlarmCounter(hrtc, counter_alarm) != HAL_OK)
842 {
843 return HAL_ERROR;
844 }
845 }
846 else
847 {
848 /* Alarm already occurred. Set it to reset values to avoid unexpected expiration */
849 if (RTC_WriteAlarmCounter(hrtc, counter_alarm) != HAL_OK)
850 {
851 return HAL_ERROR;
852 }
853 }
854
855 /* Update date */
856 RTC_DateUpdate(hrtc, days_elapsed);
857 }
858 else
859 {
860 sTime->Hours = hours;
861 }
862
863 /* Check the input parameters format */
864 if (Format != RTC_FORMAT_BIN)
865 {
866 /* Convert the time structure parameters to BCD format */
867 sTime->Hours = (uint8_t)RTC_ByteToBcd2(sTime->Hours);
868 sTime->Minutes = (uint8_t)RTC_ByteToBcd2(sTime->Minutes);
869 sTime->Seconds = (uint8_t)RTC_ByteToBcd2(sTime->Seconds);
870 }
871
872 return HAL_OK;
873}
