149HAL_StatusTypeDef HAL_IWDG_Init(IWDG_HandleTypeDef *hiwdg)
150{
151 uint32_t tickstart;
152
153 /* Check the IWDG handle allocation */
154 if (hiwdg == NULL)
155 {
156 return HAL_ERROR;
157 }
158
159 /* Check the parameters */
160 assert_param(IS_IWDG_ALL_INSTANCE(hiwdg->Instance));
161 assert_param(IS_IWDG_PRESCALER(hiwdg->Init.Prescaler));
162 assert_param(IS_IWDG_RELOAD(hiwdg->Init.Reload));
163
164 /* Enable IWDG. LSI requires ready */
165 __HAL_IWDG_START(hiwdg);
166
167 /* Enable write access to IWDG_PR and IWDG_RLR registers by writing 0x5555 in KR */
168 IWDG_ENABLE_WRITE_ACCESS(hiwdg);
169
170 /* Write to IWDG registers the Prescaler & Reload values to work with */
171 hiwdg->Instance->PR = hiwdg->Init.Prescaler;
172 hiwdg->Instance->RLR = hiwdg->Init.Reload;
173
174 /* Check pending flag, if previous update not done, return timeout */
175 tickstart = HAL_GetTick();
176
177 /* Wait for register to be updated */
178 while (hiwdg->Instance->SR != RESET)
179 {
180 if ((HAL_GetTick() - tickstart) > HAL_IWDG_DEFAULT_TIMEOUT)
181 {
182 return HAL_TIMEOUT;
183 }
184 }
185 /* Reload IWDG counter with value defined in the reload register */
186 __HAL_IWDG_RELOAD_COUNTER(hiwdg);
187
188 /* Return function status */
189 return HAL_OK;
190}
