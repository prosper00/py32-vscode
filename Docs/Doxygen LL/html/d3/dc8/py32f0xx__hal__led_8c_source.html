158HAL_StatusTypeDef HAL_LED_SetComDisplay(LED_HandleTypeDef *hled, uint8_t comCh, uint8_t data)
159{
160 uint32_t position=0x00U;
161 uint32_t chcurrent = 0x00U;
162 uint32_t *pTmp;
163
164 if (hled->State == HAL_LED_STATE_READY)
165 {
166 /* Process Locked */
167 __HAL_LOCK(hled);
168 hled->State = HAL_LED_STATE_BUSY;
169
170 while ((comCh >> position) != 0)
171 {
172 /* Get the COM channel position */
173 chcurrent = comCh & (1U << position);
174
175 if(chcurrent)
176 {
177 pTmp = ((uint32_t *)&hled->Instance->DR0) + position;
178 WRITE_REG(*pTmp, data);
179 }
180
181 position++;
182 }
183
184 /* Process Unlocked */
185 __HAL_UNLOCK(hled);
186 hled->State = HAL_LED_STATE_READY;
187
188 return HAL_OK;
189 }

203{

206 tmpreg=hled->Init.ComDrive | (hled->Init.ComNum << LED_CR_LED_COM_SEL_Pos);
207 MODIFY_REG(hled->Instance->CR,
