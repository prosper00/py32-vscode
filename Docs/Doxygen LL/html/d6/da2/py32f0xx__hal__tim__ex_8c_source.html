134HAL_StatusTypeDef HAL_TIMEx_HallSensor_Init(TIM_HandleTypeDef *htim, TIM_HallSensor_InitTypeDef *sConfig)
135{
136 TIM_OC_InitTypeDef OC_Config;
137
138 /* Check the TIM handle allocation */
139 if (htim == NULL)
140 {
141 return HAL_ERROR;
142 }
143
144 /* Check the parameters */
145 assert_param(IS_TIM_HALL_SENSOR_INTERFACE_INSTANCE(htim->Instance));
146 assert_param(IS_TIM_COUNTER_MODE(htim->Init.CounterMode));
147 assert_param(IS_TIM_CLOCKDIVISION_DIV(htim->Init.ClockDivision));
148 assert_param(IS_TIM_AUTORELOAD_PRELOAD(htim->Init.AutoReloadPreload));
149 assert_param(IS_TIM_IC_POLARITY(sConfig->IC1Polarity));
150 assert_param(IS_TIM_IC_PRESCALER(sConfig->IC1Prescaler));
151 assert_param(IS_TIM_IC_FILTER(sConfig->IC1Filter));
152
153 if (htim->State == HAL_TIM_STATE_RESET)
154 {
155 /* Allocate lock resource and initialize it */
156 htim->Lock = HAL_UNLOCKED;
157
158#if (USE_HAL_TIM_REGISTER_CALLBACKS == 1)
159 /* Reset interrupt callbacks to legacy week callbacks */
160 TIM_ResetCallback(htim);
161
162 if (htim->HallSensor_MspInitCallback == NULL)
163 {
164 htim->HallSensor_MspInitCallback = HAL_TIMEx_HallSensor_MspInit;
165 }
166 /* Init the low level hardware : GPIO, CLOCK, NVIC */
167 htim->HallSensor_MspInitCallback(htim);
168#else
169 /* Init the low level hardware : GPIO, CLOCK, NVIC and DMA */
170 HAL_TIMEx_HallSensor_MspInit(htim);
171#endif /* USE_HAL_TIM_REGISTER_CALLBACKS */
172 }
173
174 /* Set the TIM state */
175 htim->State = HAL_TIM_STATE_BUSY;
176
177 /* Configure the Time base in the Encoder Mode */
178 TIM_Base_SetConfig(htim->Instance, &htim->Init);
179
180 /* Configure the Channel 1 as Input Channel to interface with the three Outputs of the Hall sensor */
181 TIM_TI1_SetConfig(htim->Instance, sConfig->IC1Polarity, TIM_ICSELECTION_TRC, sConfig->IC1Filter);
182
183 /* Reset the IC1PSC Bits */
184 htim->Instance->CCMR1 &= ~TIM_CCMR1_IC1PSC;
185 /* Set the IC1PSC value */
186 htim->Instance->CCMR1 |= sConfig->IC1Prescaler;
187
188 /* Enable the Hall sensor interface (XOR function of the three inputs) */
189 htim->Instance->CR2 |= TIM_CR2_TI1S;
190
191 /* Select the TIM_TS_TI1F_ED signal as Input trigger for the TIM */
192 htim->Instance->SMCR &= ~TIM_SMCR_TS;
193 htim->Instance->SMCR |= TIM_TS_TI1F_ED;
194
195 /* Use the TIM_TS_TI1F_ED signal to reset the TIM counter each edge detection */
196 htim->Instance->SMCR &= ~TIM_SMCR_SMS;
197 htim->Instance->SMCR |= TIM_SLAVEMODE_RESET;
198
199 /* Program channel 2 in PWM 2 mode with the desired Commutation_Delay*/
200 OC_Config.OCFastMode = TIM_OCFAST_DISABLE;
201 OC_Config.OCIdleState = TIM_OCIDLESTATE_RESET;
202 OC_Config.OCMode = TIM_OCMODE_PWM2;
203 OC_Config.OCNIdleState = TIM_OCNIDLESTATE_RESET;
204 OC_Config.OCNPolarity = TIM_OCNPOLARITY_HIGH;
205 OC_Config.OCPolarity = TIM_OCPOLARITY_HIGH;
206 OC_Config.Pulse = sConfig->Commutation_Delay;
207
208 TIM_OC2_SetConfig(htim->Instance, &OC_Config);
209
210 /* Select OC2REF as trigger output on TRGO: write the MMS bits in the TIMx_CR2
211 register to 101 */
212 htim->Instance->CR2 &= ~TIM_CR2_MMS;
213 htim->Instance->CR2 |= TIM_TRGO_OC2REF;
214
215 /* Initialize the TIM state*/
216 htim->State = HAL_TIM_STATE_READY;
217
218 return HAL_OK;
219}
