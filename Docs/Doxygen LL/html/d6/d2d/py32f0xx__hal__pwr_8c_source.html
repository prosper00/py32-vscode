245HAL_StatusTypeDef HAL_PWR_ConfigPVD(PWR_PVDTypeDef *sConfigPVD)
246{
247 /* Check the parameters */
248 assert_param(IS_PWR_PVD_LEVEL(sConfigPVD->PVDLevel));
249 assert_param(IS_PWR_PVD_MODE(sConfigPVD->Mode));
250
251 /* Set PVD level bits only according to PVDLevel value */
252 MODIFY_REG(PWR->CR2, (PWR_CR2_PVDT | PWR_CR2_FLTEN | PWR_CR2_FLT_TIME | PWR_CR2_SRCSEL), \
253 (sConfigPVD->PVDLevel | sConfigPVD->PVDFilter | sConfigPVD->PVDSource));
254
255 /* Clear any previous config, in case no event or IT mode is selected */
256 __HAL_PWR_PVD_EXTI_DISABLE_EVENT();
257 __HAL_PWR_PVD_EXTI_DISABLE_IT();
258 __HAL_PWR_PVD_EXTI_DISABLE_FALLING_EDGE();
259 __HAL_PWR_PVD_EXTI_DISABLE_RISING_EDGE();
260
261 /* Configure interrupt mode */
262 if((sConfigPVD->Mode & PVD_MODE_IT) == PVD_MODE_IT)
263 {
264 __HAL_PWR_PVD_EXTI_ENABLE_IT();
265 }
266
267 /* Configure event mode */
268 if((sConfigPVD->Mode & PVD_MODE_EVT) == PVD_MODE_EVT)
269 {
270 __HAL_PWR_PVD_EXTI_ENABLE_EVENT();
271 }
272
273 /* Configure the edge */
274 if((sConfigPVD->Mode & PVD_RISING_EDGE) == PVD_RISING_EDGE)
275 {
276 __HAL_PWR_PVD_EXTI_ENABLE_RISING_EDGE();
277 }
278
279 if((sConfigPVD->Mode & PVD_FALLING_EDGE) == PVD_FALLING_EDGE)
280 {
281 __HAL_PWR_PVD_EXTI_ENABLE_FALLING_EDGE();
282 }
283
284 return HAL_OK;
285}

373void HAL_PWR_EnterSLEEPMode(uint8_t SLEEPEntry)
374{
375 /* Check the parameters */
376 assert_param(IS_PWR_SLEEP_ENTRY(SLEEPEntry));
377
378 /* Clear SLEEPDEEP bit of Cortex System Control Register */
379 CLEAR_BIT(SCB->SCR, ((uint32_t)SCB_SCR_SLEEPDEEP_Msk));
380
381 /* Select SLEEP mode entry -------------------------------------------------*/
382 if(SLEEPEntry == PWR_SLEEPENTRY_WFI)
383 {
384 /* Request Wait For Interrupt */
385 __WFI();
386 }
387 else
388 {
389 /* Request Wait For Event */
390 __SEV();
391 __WFE();
392 __WFE();
393 }
394}

430void HAL_PWR_EnterSTOPMode(uint32_t Regulator, uint8_t STOPEntry)
431{
432 /* Check the parameters */
433 assert_param(IS_PWR_REGULATOR(Regulator));
434 assert_param(IS_PWR_STOP_ENTRY(STOPEntry));
435
436 if (Regulator != PWR_MAINREGULATOR_ON)
437 {
438 /* Stop mode with Low-Power Regulator */
439 SET_BIT(PWR->CR1,PWR_CR1_LPR);
440 }
441 else
442 {
443 /* Stop mode with Main Regulator */
444 CLEAR_BIT(PWR->CR1,PWR_CR1_LPR);
445 }
446
447 /* Set SLEEPDEEP bit of Cortex System Control Register */
448 SET_BIT(SCB->SCR, ((uint32_t)SCB_SCR_SLEEPDEEP_Msk));
449
450 /* Select Stop mode entry --------------------------------------------------*/
451 if(STOPEntry == PWR_STOPENTRY_WFI)
452 {
453 /* Request Wait For Interrupt */
454 __WFI();
455 }
456 else
457 {
458 /* Request Wait For Event */
459 __SEV();
460 __WFE();
461 __WFE();
462 }
463
464 /* Reset SLEEPDEEP bit of Cortex System Control Register */
465 CLEAR_BIT(SCB->SCR, ((uint32_t)SCB_SCR_SLEEPDEEP_Msk));
466}
