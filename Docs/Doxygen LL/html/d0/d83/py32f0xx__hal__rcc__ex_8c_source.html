103HAL_StatusTypeDef HAL_RCCEx_PeriphCLKConfig(RCC_PeriphCLKInitTypeDef *PeriphClkInit)
104{
105#if defined(RCC_BDCR_RTCSEL)
106 uint32_t tickstart = 0U, temp_reg = 0U;
107 FlagStatus pwrclkchanged = RESET;
108#endif
109 /* Check the parameters */
110 assert_param(IS_RCC_PERIPHCLOCK(PeriphClkInit->PeriphClockSelection));
111#if defined(RCC_BDCR_RTCSEL)
112 /*------------------------------- RTC Configuration ------------------------*/
113 if ((((PeriphClkInit->PeriphClockSelection) & RCC_PERIPHCLK_RTC) == RCC_PERIPHCLK_RTC))
114 {
115 /* check for RTC Parameters used to output RTCCLK */
116 assert_param(IS_RCC_RTCCLKSOURCE(PeriphClkInit->RTCClockSelection));
117
118 /* As soon as function is called to change RTC clock source, activation of the
119 power domain is done. */
120 /* Requires to enable write access to Backup Domain of necessary */
121 if (__HAL_RCC_PWR_IS_CLK_DISABLED())
122 {
123 __HAL_RCC_PWR_CLK_ENABLE();
124 pwrclkchanged = SET;
125 }
126
127 if (HAL_IS_BIT_CLR(PWR->CR1, PWR_CR1_DBP))
128 {
129 /* Enable write access to Backup domain */
130 SET_BIT(PWR->CR1, PWR_CR1_DBP);
131
132 /* Wait for Backup domain Write protection disable */
133 tickstart = HAL_GetTick();
134
135 while (HAL_IS_BIT_CLR(PWR->CR1, PWR_CR1_DBP))
136 {
137 if ((HAL_GetTick() - tickstart) > RCC_DBP_TIMEOUT_VALUE)
138 {
139 return HAL_TIMEOUT;
140 }
141 }
142 }
143
144 /* Reset the Backup domain only if the RTC Clock source selection is modified from reset value */
145 temp_reg = (RCC->BDCR & RCC_BDCR_RTCSEL);
146 if ((temp_reg != 0x00000000U) && (temp_reg != (PeriphClkInit->RTCClockSelection & RCC_BDCR_RTCSEL)))
147 {
148 /* Store the content of BDCR register before the reset of Backup Domain */
149 temp_reg = (RCC->BDCR & ~(RCC_BDCR_RTCSEL));
150 /* RTC Clock selection can be changed only if the Backup Domain is reset */
151 __HAL_RCC_BACKUPRESET_FORCE();
152 __HAL_RCC_BACKUPRESET_RELEASE();
153 /* Restore the Content of BDCR register */
154 RCC->BDCR = temp_reg;
155
156#if defined(RCC_LSE_SUPPORT)
157 /* Wait for LSERDY if LSE was enabled */
158 if (HAL_IS_BIT_SET(temp_reg, RCC_BDCR_LSEON))
159 {
160 /* Get Start Tick */
161 tickstart = HAL_GetTick();
162
163 /* Wait till LSE is ready */
164 while (__HAL_RCC_GET_FLAG(RCC_FLAG_LSERDY) == RESET)
165 {
166 if ((HAL_GetTick() - tickstart) > RCC_LSE_TIMEOUT_VALUE)
167 {
168 return HAL_TIMEOUT;
169 }
170 }
171 }
172#endif
173 }
174 __HAL_RCC_RTC_CONFIG(PeriphClkInit->RTCClockSelection);
175
176 /* Require to disable power clock if necessary */
177 if (pwrclkchanged == SET)
178 {
179 __HAL_RCC_PWR_CLK_DISABLE();
180 }
181 }
182#endif /*RCC_BDCR_RTCSEL*/
183
184#if defined(RCC_CCIPR_PVDSEL)
185 /*-------------------------- PVD clock source configuration -------------------*/

558void HAL_RCCEx_DisableLSCO(void)
559{
560 FlagStatus pwrclkchanged = RESET;
561 FlagStatus backupchanged = RESET;
562
563 /* Update LSCOEN bit in Backup Domain control register */
564 if (__HAL_RCC_PWR_IS_CLK_DISABLED())
565 {
566 __HAL_RCC_PWR_CLK_ENABLE();
567 pwrclkchanged = SET;
568 }
569 if (HAL_IS_BIT_CLR(PWR->CR1, PWR_CR1_DBP))
570 {
571 /* Enable access to the backup domain */
572 HAL_PWR_EnableBkUpAccess();
573 backupchanged = SET;
574 }
575
576 CLEAR_BIT(RCC->BDCR, RCC_BDCR_LSCOEN);
577
578 /* Restore previous configuration */
579 if (backupchanged == SET)
580 {
581 /* Disable access to the backup domain */
582 HAL_PWR_DisableBkUpAccess();
583 }
584 if (pwrclkchanged == SET)
585 {
586 __HAL_RCC_PWR_CLK_DISABLE();
587 }
