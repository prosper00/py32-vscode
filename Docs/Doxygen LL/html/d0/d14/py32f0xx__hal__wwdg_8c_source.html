153HAL_StatusTypeDef HAL_WWDG_Init(WWDG_HandleTypeDef *hwwdg)
154{
155 /* Check the WWDG handle allocation */
156 if (hwwdg == NULL)
157 {
158 return HAL_ERROR;
159 }
160
161 /* Check the parameters */
162 assert_param(IS_WWDG_ALL_INSTANCE(hwwdg->Instance));
163 assert_param(IS_WWDG_PRESCALER(hwwdg->Init.Prescaler));
164 assert_param(IS_WWDG_WINDOW(hwwdg->Init.Window));
165 assert_param(IS_WWDG_COUNTER(hwwdg->Init.Counter));
166 assert_param(IS_WWDG_EWI_MODE(hwwdg->Init.EWIMode));
167
168#if (USE_HAL_WWDG_REGISTER_CALLBACKS == 1)
169 /* Reset Callback pointers */
170 if(hwwdg->EwiCallback == NULL)
171 {
172 hwwdg->EwiCallback = HAL_WWDG_EarlyWakeupCallback;
173 }
174
175 if(hwwdg->MspInitCallback == NULL)
176 {
177 hwwdg->MspInitCallback = HAL_WWDG_MspInit;
178 }
179
180 /* Init the low level hardware */
181 hwwdg->MspInitCallback(hwwdg);
182#else
183 /* Init the low level hardware */
184 HAL_WWDG_MspInit(hwwdg);
185#endif
186
187 /* Set WWDG Counter */
188 WRITE_REG(hwwdg->Instance->CR, (WWDG_CR_WDGA | hwwdg->Init.Counter));
189 /* Set WWDG Prescaler and Window */
190 WRITE_REG(hwwdg->Instance->CFR, (hwwdg->Init.EWIMode | hwwdg->Init.Prescaler | hwwdg->Init.Window));
191
192 /* Return function status */
193 return HAL_OK;
194}

340void HAL_WWDG_IRQHandler(WWDG_HandleTypeDef *hwwdg)
341{
342 /* Check if Early Wakeup Interrupt is enable */
343 if (__HAL_WWDG_GET_IT_SOURCE(hwwdg, WWDG_IT_EWI) != RESET)
344 {
345 /* Check if WWDG Early Wakeup Interrupt occurred */
346 if (__HAL_WWDG_GET_FLAG(hwwdg, WWDG_FLAG_EWIF) != RESET)
347 {
348 /* Clear the WWDG Early Wakeup flag */
349 __HAL_WWDG_CLEAR_FLAG(hwwdg, WWDG_FLAG_EWIF);
350
351#if (USE_HAL_WWDG_REGISTER_CALLBACKS == 1)
352 /* Early Wakeup registered callback */
353 hwwdg->EwiCallback(hwwdg);
354#else
355 /* Early Wakeup callback */
356 HAL_WWDG_EarlyWakeupCallback(hwwdg);
357#endif
358 }
359 }
