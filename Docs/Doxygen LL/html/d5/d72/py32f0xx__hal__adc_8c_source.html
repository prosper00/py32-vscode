186HAL_StatusTypeDef HAL_ADC_Init(ADC_HandleTypeDef* hadc)
187{
188 HAL_StatusTypeDef tmp_hal_status = HAL_OK;
189 uint32_t tmpCFGR1 = 0U;
190
191 /* Check ADC handle */
192 if(hadc == NULL)
193 {
194 return HAL_ERROR;
195 }
196
197 /* Check the parameters */
198 assert_param(IS_ADC_ALL_INSTANCE(hadc->Instance));
199 assert_param(IS_ADC_CLOCKPRESCALER(hadc->Init.ClockPrescaler));
200 assert_param(IS_ADC_RESOLUTION(hadc->Init.Resolution));
201 assert_param(IS_ADC_DATA_ALIGN(hadc->Init.DataAlign));
202 assert_param(IS_ADC_SCAN_MODE(hadc->Init.ScanConvMode));
203 assert_param(IS_FUNCTIONAL_STATE(hadc->Init.ContinuousConvMode));
204 assert_param(IS_FUNCTIONAL_STATE(hadc->Init.DiscontinuousConvMode));
205 assert_param(IS_ADC_EXTTRIG_EDGE(hadc->Init.ExternalTrigConvEdge));
206 assert_param(IS_ADC_EXTTRIG(hadc->Init.ExternalTrigConv));
207#if (defined(DMA) || defined(DMA1))
208 assert_param(IS_FUNCTIONAL_STATE(hadc->Init.DMAContinuousRequests));
209#endif
210 assert_param(IS_ADC_EOC_SELECTION(hadc->Init.EOCSelection));
211 assert_param(IS_ADC_OVERRUN(hadc->Init.Overrun));
212 assert_param(IS_FUNCTIONAL_STATE(hadc->Init.LowPowerAutoWait));
213
214 /* As prerequisite, into HAL_ADC_MspInit(), ADC clock must be configured */
215 /* at RCC top level depending on both possible clock sources: */
216 /* APB clock or HSI clock. */
217 /* Refer to header of this file for more details on clock enabling procedure*/
218
219 /* Actions performed only if ADC is coming from state reset: */
220 /* - Initialization of ADC MSP */
221 /* - ADC voltage regulator enable */
222 if (hadc->State == HAL_ADC_STATE_RESET)
223 {
224 /* Initialize ADC error code */
225 ADC_CLEAR_ERRORCODE(hadc);
226
227 /* Allocate lock resource and initialize it */
228 hadc->Lock = HAL_UNLOCKED;
229
230#if (USE_HAL_ADC_REGISTER_CALLBACKS == 1)
231 /* Init the ADC Callback settings */
232 hadc->ConvCpltCallback = HAL_ADC_ConvCpltCallback; /* Legacy weak callback */
233 hadc->ConvHalfCpltCallback = HAL_ADC_ConvHalfCpltCallback; /* Legacy weak callback */
234 hadc->LevelOutOfWindowCallback = HAL_ADC_LevelOutOfWindowCallback; /* Legacy weak callback */
235 hadc->ErrorCallback = HAL_ADC_ErrorCallback; /* Legacy weak callback */
236
237 if (hadc->MspInitCallback == NULL)
238 {
239 hadc->MspInitCallback = HAL_ADC_MspInit; /* Legacy weak MspInit */
240 }
241
242 /* Init the low level hardware */
243 hadc->MspInitCallback(hadc);
244#else
245 /* Init the low level hardware */
246 HAL_ADC_MspInit(hadc);
247#endif /* USE_HAL_ADC_REGISTER_CALLBACKS */
248 }
249
250 /* Configuration of ADC parameters if previous preliminary actions are */
251 /* correctly completed. */
252 /* and if there is no conversion on going on regular group (ADC can be */
253 /* enabled anyway, in case of call of this function to update a parameter */
254 /* on the fly). */
255 if (HAL_IS_BIT_CLR(hadc->State, HAL_ADC_STATE_ERROR_INTERNAL) &&
256 (tmp_hal_status == HAL_OK) &&
257 (ADC_IS_CONVERSION_ONGOING_REGULAR(hadc) == RESET) )
258 {
259 /* Set ADC state */
260 ADC_STATE_CLR_SET(hadc->State,
261 HAL_ADC_STATE_REG_BUSY,
262 HAL_ADC_STATE_BUSY_INTERNAL);
263
264 /* Parameters update conditioned to ADC state: */
265 /* Parameters that can be updated only when ADC is disabled: */
266 /* - ADC clock mode */
267 /* - ADC clock prescaler */
268 /* - ADC resolution */
269 if (ADC_IS_ENABLE(hadc) == RESET)
270 {
271 /* Some parameters of this register are not reset, since they are set */
272 /* by other functions and must be kept in case of usage of this */
273 /* function on the fly (update of a parameter of ADC_InitTypeDef */
274 /* without needing to reconfigure all other ADC groups/channels */
275 /* parameters): */
276 /* - internal measurement paths: temperature sensor, Vref */
277 /* (set into HAL_ADC_ConfigChannel() ) */
278
279 /* Configuration of ADC resolution */
280 MODIFY_REG(hadc->Instance->CFGR1,
281 ADC_CFGR1_RESSEL,
282 hadc->Init.Resolution );
283
284 /* Configuration of ADC clock mode: clock source AHB or HSI with */
285 /* selectable prescaler */
286 MODIFY_REG(hadc->Instance->CFGR2,
287 ADC_CFGR2_CKMODE,
288 hadc->Init.ClockPrescaler );
289 }
290
291 /* Configuration of ADC: */
292 /* - discontinuous mode */
293 /* - LowPowerAutoWait mode */
294 /* - continuous conversion mode */
295 /* - overrun */
296 /* - external trigger to start conversion */
297 /* - external trigger polarity */
298 /* - data alignment */
299 /* - resolution */
300 /* - scan direction */
301 /* - DMA continuous request */
302#if (defined(DMA) || defined(DMA1))
303 hadc->Instance->CFGR1 &= ~( ADC_CFGR1_DISCEN |
304 ADC_CFGR1_CONT |
305 ADC_CFGR1_OVRMOD |
306 ADC_CFGR1_EXTSEL |
307 ADC_CFGR1_EXTEN |
308 ADC_CFGR1_ALIGN |
309 ADC_CFGR1_SCANDIR |
310 ADC_CFGR1_DMACFG );
311
312 tmpCFGR1 |= (ADC_CFGR1_AUTOWAIT((uint32_t)hadc->Init.LowPowerAutoWait) |
313 ADC_CFGR1_CONTINUOUS((uint32_t)hadc->Init.ContinuousConvMode) |
314 ADC_CFGR1_OVERRUN(hadc->Init.Overrun) |
315 hadc->Init.DataAlign |
316 ADC_SCANDIR(hadc->Init.ScanConvMode) |
317 ADC_CFGR1_DMACONTREQ((uint32_t)hadc->Init.DMAContinuousRequests) );
318#else
319 hadc->Instance->CFGR1 &= ~( ADC_CFGR1_DISCEN |
320 ADC_CFGR1_CONT |
321 ADC_CFGR1_OVRMOD |
322 ADC_CFGR1_EXTSEL |
323 ADC_CFGR1_EXTEN |
324 ADC_CFGR1_ALIGN |
325 ADC_CFGR1_SCANDIR );
326
327 tmpCFGR1 |= (ADC_CFGR1_AUTOWAIT((uint32_t)hadc->Init.LowPowerAutoWait) |
328 ADC_CFGR1_CONTINUOUS((uint32_t)hadc->Init.ContinuousConvMode) |
329 ADC_CFGR1_OVERRUN(hadc->Init.Overrun) |
330 hadc->Init.DataAlign |
331 ADC_SCANDIR(hadc->Init.ScanConvMode) );
332#endif
333
334 /* Enable discontinuous mode only if continuous mode is disabled */
335 if (hadc->Init.DiscontinuousConvMode == ENABLE)
336 {
337 if (hadc->Init.ContinuousConvMode == DISABLE)
338 {
339 /* Enable the selected ADC group regular discontinuous mode */
340 tmpCFGR1 |= ADC_CFGR1_DISCEN;
341 }
342 else
343 {
344 /* ADC regular group discontinuous was intended to be enabled, */
345 /* but ADC regular group modes continuous and sequencer discontinuous */
346 /* cannot be enabled simultaneously. */
347
348 /* Update ADC state machine to error */
349 SET_BIT(hadc->State, HAL_ADC_STATE_ERROR_CONFIG);
350
351 /* Set ADC error code to ADC IP internal error */
352 SET_BIT(hadc->ErrorCode, HAL_ADC_ERROR_INTERNAL);
353 }
354 }
355
356 /* Enable external trigger if trigger selection is different of software */
357 /* start. */
358 /* Note: This configuration keeps the hardware feature of parameter */
359 /* ExternalTrigConvEdge "trigger edge none" equivalent to */
360 /* software start. */
361 if (hadc->Init.ExternalTrigConv != ADC_SOFTWARE_START)
362 {
363 tmpCFGR1 |= ( hadc->Init.ExternalTrigConv |
364 hadc->Init.ExternalTrigConvEdge );
365 }
366
367 /* Update ADC configuration register with previous settings */
368 hadc->Instance->CFGR1 |= tmpCFGR1;
369
370 /* Channel sampling time configuration */
371 /* Management of parameters "SamplingTimeCommon" and "SamplingTime" */
372 /* (obsolete): sampling time set in this function if parameter */
373 /* "SamplingTimeCommon" has been set to a valid sampling time. */
374 /* Otherwise, sampling time is set into ADC channel initialization */
375 /* structure with parameter "SamplingTime" (obsolete). */
376 if (IS_ADC_SAMPLE_TIME(hadc->Init.SamplingTimeCommon))
377 {
378 /* Channel sampling time configuration */
379 /* Clear the old sample time */
380 hadc->Instance->SMPR &= ~(ADC_SMPR_SMP);
381
382 /* Set the new sample time */
383 hadc->Instance->SMPR |= ADC_SMPR_SET(hadc->Init.SamplingTimeCommon);
384 }
385
386 /* Check back that ADC registers have effectively been configured to */
387 /* ensure of no potential problem of ADC core IP clocking. */
388 /* Check through register CFGR1 (excluding analog watchdog configuration: */
389 /* set into separate dedicated function, and bits of ADC resolution set */
390 /* out of temporary variable 'tmpCFGR1'). */
391 if ((hadc->Instance->CFGR1 & ~(ADC_CFGR1_AWDCH | ADC_CFGR1_AWDEN | ADC_CFGR1_AWDSGL | ADC_CFGR1_RESSEL))
392 == tmpCFGR1)
393 {
394 /* Set ADC error code to none */
395 ADC_CLEAR_ERRORCODE(hadc);
396
397 /* Set the ADC state */
398 ADC_STATE_CLR_SET(hadc->State,
399 HAL_ADC_STATE_BUSY_INTERNAL,
400 HAL_ADC_STATE_READY);
401 }
402 else
403 {
404 /* Update ADC state machine to error */
405 ADC_STATE_CLR_SET(hadc->State,
406 HAL_ADC_STATE_BUSY_INTERNAL,
407 HAL_ADC_STATE_ERROR_INTERNAL);
408
409 /* Set ADC error code to ADC IP internal error */
410 SET_BIT(hadc->ErrorCode, HAL_ADC_ERROR_INTERNAL);
411
412 tmp_hal_status = HAL_ERROR;
413 }
414
415 }
416 else
417 {
418 /* Update ADC state machine to error */
419 SET_BIT(hadc->State, HAL_ADC_STATE_ERROR_INTERNAL);
420
421 tmp_hal_status = HAL_ERROR;
422 }
423
424 /* Return function status */
425 return tmp_hal_status;
426}

440HAL_StatusTypeDef HAL_ADC_DeInit(ADC_HandleTypeDef* hadc)
441{
442 HAL_StatusTypeDef tmp_hal_status = HAL_OK;
443
444 /* Check ADC handle */
445 if(hadc == NULL)
446 {
447 return HAL_ERROR;
448 }
449
450 /* Check the parameters */
451 assert_param(IS_ADC_ALL_INSTANCE(hadc->Instance));
452
453 /* Set ADC state */
454 SET_BIT(hadc->State, HAL_ADC_STATE_BUSY_INTERNAL);
455
456 /* Stop potential conversion on going, on regular group */
457 tmp_hal_status = ADC_ConversionStop(hadc);
458
459 /* Disable ADC peripheral if conversions are effectively stopped */
460 if (tmp_hal_status == HAL_OK)
461 {
462 /* Disable the ADC peripheral */
463 tmp_hal_status = ADC_Disable(hadc);
464
465 /* Check if ADC is effectively disabled */
466 if (tmp_hal_status != HAL_ERROR)
467 {
468 /* Change ADC state */
469 hadc->State = HAL_ADC_STATE_READY;
470 }
471 }
472
473 /* Reset ADC peripheral */
474 __HAL_RCC_ADC_FORCE_RESET();
475 __HAL_RCC_ADC_RELEASE_RESET();
476
477 /* Configuration of ADC parameters if previous preliminary actions are */
478 /* correctly completed. */
479 if (tmp_hal_status != HAL_ERROR)
480 {
481
482 /* ========== Reset ADC registers ========== */
483 /* Reset register IER */
484 __HAL_ADC_DISABLE_IT(hadc, (ADC_IT_AWD | ADC_IT_OVR |
485 ADC_IT_EOS | ADC_IT_EOC |
486 ADC_IT_EOSMP /*| ADC_IT_RDY*/ ) );
487
488 /* Reset register ISR */
489 __HAL_ADC_CLEAR_FLAG(hadc, (ADC_FLAG_AWD | ADC_FLAG_OVR |
490 ADC_FLAG_EOS | ADC_FLAG_EOC |
491 ADC_FLAG_EOSMP /*| ADC_FLAG_RDY*/ ) );
492
493 /* Reset register CR */
494 /* Bits ADC_CR_ADCAL, ADC_CR_ADSTP, ADC_CR_ADSTART are in access mode */
495 /* "read-set": no direct reset applicable. */
496
497 /* Reset register CFGR1 */
498#if (defined(DMA) || defined(DMA1))
499 hadc->Instance->CFGR1 &= ~(ADC_CFGR1_AWDCH | ADC_CFGR1_AWDEN | ADC_CFGR1_AWDSGL | ADC_CFGR1_DISCEN |
500 /*ADC_CFGR1_AUTOFF |*/ ADC_CFGR1_WAIT | ADC_CFGR1_CONT | ADC_CFGR1_OVRMOD |
501 ADC_CFGR1_EXTEN | ADC_CFGR1_EXTSEL | ADC_CFGR1_ALIGN | ADC_CFGR1_RESSEL |
502 ADC_CFGR1_SCANDIR | ADC_CFGR1_DMACFG | ADC_CFGR1_DMAEN );
503#else
504 hadc->Instance->CFGR1 &= ~(ADC_CFGR1_AWDCH | ADC_CFGR1_AWDEN | ADC_CFGR1_AWDSGL | ADC_CFGR1_DISCEN |
505 /*ADC_CFGR1_AUTOFF |*/ ADC_CFGR1_WAIT | ADC_CFGR1_CONT | ADC_CFGR1_OVRMOD |
506 ADC_CFGR1_EXTEN | ADC_CFGR1_EXTSEL | ADC_CFGR1_ALIGN | ADC_CFGR1_RESSEL |
507 ADC_CFGR1_SCANDIR );
508#endif
509
510 /* Reset register CFGR2 */
511 /* Note: Update of ADC clock mode is conditioned to ADC state disabled: */
512 /* already done above. */
513 hadc->Instance->CFGR2 &= ~ADC_CFGR2_CKMODE;
514
515 /* Reset register SMPR */
516 hadc->Instance->SMPR &= ~ADC_SMPR_SMP;
517
518 /* Reset register TR */
519 hadc->Instance->TR &= ~(ADC_TR_HT | ADC_TR_LT);
520
521 /* Reset register CHSELR */
522 hadc->Instance->CHSELR &= ~(ADC_CHSELR_CHSEL12 | ADC_CHSELR_CHSEL11 | ADC_CHSELR_CHSEL9 | ADC_CHSELR_CHSEL8 |
523 ADC_CHSELR_CHSEL7 | ADC_CHSELR_CHSEL6 | ADC_CHSELR_CHSEL5 | ADC_CHSELR_CHSEL4 |
524 ADC_CHSELR_CHSEL3 | ADC_CHSELR_CHSEL2 | ADC_CHSELR_CHSEL1 | ADC_CHSELR_CHSEL0 );
525
526 /* Reset register DR */
527 /* bits in access mode read only, no direct reset applicable*/
528
529 /* Reset register CCR */
530 ADC->CCR &= ~(ADC_CCR_ALL);
531
532 /* ========== Hard reset ADC peripheral ========== */
533 /* Performs a global reset of the entire ADC peripheral: ADC state is */
534 /* forced to a similar state after device power-on. */
535 /* If needed, copy-paste and uncomment the following reset code into */
536 /* function "void HAL_ADC_MspInit(ADC_HandleTypeDef* hadc)": */
537 /* */
538 /* __HAL_RCC_ADC1_FORCE_RESET() */
539 /* __HAL_RCC_ADC1_RELEASE_RESET() */
540
541#if (USE_HAL_ADC_REGISTER_CALLBACKS == 1)
542 if (hadc->MspDeInitCallback == NULL)
543 {
544 hadc->MspDeInitCallback = HAL_ADC_MspDeInit; /* Legacy weak MspDeInit */
545 }
546
547 /* DeInit the low level hardware */
548 hadc->MspDeInitCallback(hadc);
549#else
550 /* DeInit the low level hardware */
551 HAL_ADC_MspDeInit(hadc);
552#endif /* USE_HAL_ADC_REGISTER_CALLBACKS */
553
554 /* Set ADC error code to none */
555 ADC_CLEAR_ERRORCODE(hadc);
556
557 /* Set ADC state */
558 hadc->State = HAL_ADC_STATE_RESET;
559 }
560
561 /* Process unlocked */
562 __HAL_UNLOCK(hadc);
563
564 /* Return function status */
565 return tmp_hal_status;
566}

827HAL_StatusTypeDef HAL_ADC_Start(ADC_HandleTypeDef* hadc)
828{
829 HAL_StatusTypeDef tmp_hal_status = HAL_OK;
830
831 /* Check the parameters */
832 assert_param(IS_ADC_ALL_INSTANCE(hadc->Instance));
833
834 /* Perform ADC enable and conversion start if no conversion is on going */
835 if (ADC_IS_CONVERSION_ONGOING_REGULAR(hadc) == RESET)
836 {
837 /* Process locked */
838 __HAL_LOCK(hadc);
839
840 /* Enable the ADC peripheral */
841 tmp_hal_status = ADC_Enable(hadc);
842
843 /* Start conversion if ADC is effectively enabled */
844 if (tmp_hal_status == HAL_OK)
845 {
846 /* Set ADC state */
847 /* - Clear state bitfield related to regular group conversion results */
848 /* - Set state bitfield related to regular operation */
849 ADC_STATE_CLR_SET(hadc->State,
850 HAL_ADC_STATE_READY | HAL_ADC_STATE_REG_EOC | HAL_ADC_STATE_REG_OVR | HAL_ADC_STATE_REG_EOSMP,
851 HAL_ADC_STATE_REG_BUSY);
852
853 /* Reset ADC all error code fields */
854 ADC_CLEAR_ERRORCODE(hadc);
855
856 /* Process unlocked */
857 /* Unlock before starting ADC conversions: in case of potential */
858 /* interruption, to let the process to ADC IRQ Handler. */
859 __HAL_UNLOCK(hadc);
860
861 /* Clear regular group conversion flag and overrun flag */
862 /* (To ensure of no unknown state from potential previous ADC */
863 /* operations) */
864 __HAL_ADC_CLEAR_FLAG(hadc, (ADC_FLAG_EOC | ADC_FLAG_EOS | ADC_FLAG_OVR));
865
866 /* Enable conversion of regular group. */
867 /* If software start has been selected, conversion starts immediately. */
868 /* If external trigger has been selected, conversion will start at next */
869 /* trigger event. */
870 hadc->Instance->CR |= ADC_CR_ADSTART;
871 }
872 }
873 else
874 {
875 tmp_hal_status = HAL_BUSY;
876 }
877
878 /* Return function status */
879 return tmp_hal_status;
880}

887HAL_StatusTypeDef HAL_ADC_Stop(ADC_HandleTypeDef* hadc)
888{
889 HAL_StatusTypeDef tmp_hal_status = HAL_OK;
890
891 /* Check the parameters */
892 assert_param(IS_ADC_ALL_INSTANCE(hadc->Instance));
893
894 /* Process locked */
895 __HAL_LOCK(hadc);
896
897 /* 1. Stop potential conversion on going, on regular group */
898 tmp_hal_status = ADC_ConversionStop(hadc);
899
900 /* Disable ADC peripheral if conversions are effectively stopped */
901 if (tmp_hal_status == HAL_OK)
902 {
903 /* 2. Disable the ADC peripheral */
904 tmp_hal_status = ADC_Disable(hadc);
905
906 /* Check if ADC is effectively disabled */
907 if (tmp_hal_status == HAL_OK)
908 {
909 /* Set ADC state */
910 ADC_STATE_CLR_SET(hadc->State,
911 HAL_ADC_STATE_REG_BUSY,
912 HAL_ADC_STATE_READY);
913 }
914 }
915

943HAL_StatusTypeDef HAL_ADC_PollForConversion(ADC_HandleTypeDef* hadc, uint32_t Timeout)
944{
945 uint32_t tickstart;
946 uint32_t tmp_Flag_EOC;
947
948 /* Check the parameters */
949 assert_param(IS_ADC_ALL_INSTANCE(hadc->Instance));
950
951 /* If end of conversion selected to end of sequence */
952 if (hadc->Init.EOCSelection == ADC_EOC_SEQ_CONV)
953 {
954 tmp_Flag_EOC = ADC_FLAG_EOS;
955 }
956 /* If end of conversion selected to end of each conversion */
957 else /* ADC_EOC_SINGLE_CONV */
958 {
959 /* Verification that ADC configuration is compliant with polling for */
960 /* each conversion: */
961 /* Particular case is ADC configured in DMA mode and ADC sequencer with */
962 /* several ranks and polling for end of each conversion. */
963 /* For code simplicity sake, this particular case is generalized to */
964 /* ADC configured in DMA mode and and polling for end of each conversion. */
965#if (defined(DMA) || defined(DMA1))
966 if (HAL_IS_BIT_SET(hadc->Instance->CFGR1, ADC_CFGR1_DMAEN))
967 {
968 /* Update ADC state machine to error */
969 SET_BIT(hadc->State, HAL_ADC_STATE_ERROR_CONFIG);
970
971 /* Process unlocked */
972 __HAL_UNLOCK(hadc);
973
974 return HAL_ERROR;
975 }
976 else
977 {
978#endif
979 tmp_Flag_EOC = (ADC_FLAG_EOC | ADC_FLAG_EOS);
980#if (defined(DMA) || defined(DMA1))
981 }
982#endif
983 }
984
985 /* Get tick count */
986 tickstart = HAL_GetTick();
987
988 /* Wait until End of Conversion flag is raised */
989 while(HAL_IS_BIT_CLR(hadc->Instance->ISR, tmp_Flag_EOC))
990 {
991 /* Check if timeout is disabled (set to infinite wait) */
992 if(Timeout != HAL_MAX_DELAY)
993 {
994 if((Timeout == 0) || ((HAL_GetTick()-tickstart) > Timeout))
995 {
996 /* Update ADC state machine to timeout */
997 SET_BIT(hadc->State, HAL_ADC_STATE_TIMEOUT);
998
999 /* Process unlocked */
1000 __HAL_UNLOCK(hadc);
1001
1002 return HAL_TIMEOUT;
1003 }
1004 }
1005 }
1006
1007 /* Update ADC state machine */
1008 SET_BIT(hadc->State, HAL_ADC_STATE_REG_EOC);
1009
1010 /* Determine whether any further conversion upcoming on group regular */
1011 /* by external trigger, continuous mode or scan sequence on going. */
1012 if(ADC_IS_SOFTWARE_START_REGULAR(hadc) &&
1013 (hadc->Init.ContinuousConvMode == DISABLE) )
1014 {
1015 /* If End of Sequence is reached, disable interrupts */
1016 if( __HAL_ADC_GET_FLAG(hadc, ADC_FLAG_EOS) )
1017 {
1018 /* Allowed to modify bits ADC_IT_EOC/ADC_IT_EOS only if bit */
1019 /* ADSTART==0 (no conversion on going) */
1020 if (ADC_IS_CONVERSION_ONGOING_REGULAR(hadc) == RESET)
1021 {
1022 /* Disable ADC end of single conversion interrupt on group regular */
1023 /* Note: Overrun interrupt was enabled with EOC interrupt in */
1024 /* HAL_Start_IT(), but is not disabled here because can be used */
1025 /* by overrun IRQ process below. */
1026 __HAL_ADC_DISABLE_IT(hadc, ADC_IT_EOC | ADC_IT_EOS);
1027
1028 /* Set ADC state */
1029 ADC_STATE_CLR_SET(hadc->State,
1030 HAL_ADC_STATE_REG_BUSY,
1031 HAL_ADC_STATE_READY);
1032 }
1033 else
1034 {
1035 /* Change ADC state to error state */
1036 SET_BIT(hadc->State, HAL_ADC_STATE_ERROR_CONFIG);
1037
1038 /* Set ADC error code to ADC IP internal error */
1039 SET_BIT(hadc->ErrorCode, HAL_ADC_ERROR_INTERNAL);
1040 }
1041 }
1042 }
1043
1044 /* Clear end of conversion flag of regular group if low power feature */
1045 /* "LowPowerAutoWait " is disabled, to not interfere with this feature */
1046 /* until data register is read using function HAL_ADC_GetValue(). */
1047 if (hadc->Init.LowPowerAutoWait == DISABLE)
1048 {
1049 /* Clear regular group conversion flag */
1050 __HAL_ADC_CLEAR_FLAG(hadc, (ADC_FLAG_EOC | ADC_FLAG_EOS));
1051 }
1052
1053 /* Return ADC state */
1054 return HAL_OK;
1055}

1461void HAL_ADC_IRQHandler(ADC_HandleTypeDef* hadc)
1462{

1464 assert_param(IS_ADC_ALL_INSTANCE(hadc->Instance));
1465 assert_param(IS_FUNCTIONAL_STATE(hadc->Init.ContinuousConvMode));
1466 assert_param(IS_ADC_EOC_SELECTION(hadc->Init.EOCSelection));
1467
1468 /* ========== Check End of Conversion flag for regular group ========== */
1469 if( (__HAL_ADC_GET_FLAG(hadc, ADC_FLAG_EOC) && __HAL_ADC_GET_IT_SOURCE(hadc, ADC_IT_EOC)) ||
1470 (__HAL_ADC_GET_FLAG(hadc, ADC_FLAG_EOS) && __HAL_ADC_GET_IT_SOURCE(hadc, ADC_IT_EOS)) )
1471 {
1472 /* Update state machine on conversion status if not in error state */
1473 if (HAL_IS_BIT_CLR(hadc->State, HAL_ADC_STATE_ERROR_INTERNAL))
1474 {
1475 /* Set ADC state */
1476 SET_BIT(hadc->State, HAL_ADC_STATE_REG_EOC);

1480 /* by external trigger, continuous mode or scan sequence on going. */
1481 if(ADC_IS_SOFTWARE_START_REGULAR(hadc) &&
1482 (hadc->Init.ContinuousConvMode == DISABLE) )
1483 {
1484 /* If End of Sequence is reached, disable interrupts */
1485 if( __HAL_ADC_GET_FLAG(hadc, ADC_FLAG_EOS) )
1486 {
1487 /* Allowed to modify bits ADC_IT_EOC/ADC_IT_EOS only if bit */
1488 /* ADSTART==0 (no conversion on going) */
1489 if (ADC_IS_CONVERSION_ONGOING_REGULAR(hadc) == RESET)
1490 {
1491 /* Disable ADC end of single conversion interrupt on group regular */

1494 /* by overrun IRQ process below. */

1499 HAL_ADC_STATE_REG_BUSY,
1500 HAL_ADC_STATE_READY);
1501 }
1502 else

1505 SET_BIT(hadc->State, HAL_ADC_STATE_ERROR_CONFIG);

1508 SET_BIT(hadc->ErrorCode, HAL_ADC_ERROR_INTERNAL);

1510 }
1511 }

1516#if (USE_HAL_ADC_REGISTER_CALLBACKS == 1)
1517 hadc->ConvCpltCallback(hadc);
1518#else
1519 HAL_ADC_ConvCpltCallback(hadc);
1520#endif /* USE_HAL_ADC_REGISTER_CALLBACKS */

1529 }

1539#else
1540 HAL_ADC_LevelOutOfWindowCallback(hadc);
1541#endif /* USE_HAL_ADC_REGISTER_CALLBACKS */
1542
1543 /* Clear ADC Analog watchdog flag */
1544 __HAL_ADC_CLEAR_FLAG(hadc, ADC_FLAG_AWD);
1545
1546 }
1547
1548
1549 /* ========== Check Overrun flag ========== */
1550 if(__HAL_ADC_GET_FLAG(hadc, ADC_FLAG_OVR) && __HAL_ADC_GET_IT_SOURCE(hadc, ADC_IT_OVR))

1555 /* overrun ") */

1557 /* error. */
1558#if (defined(DMA) || defined(DMA1))
1559 if ((hadc->Init.Overrun == ADC_OVR_DATA_PRESERVED) ||
1560 HAL_IS_BIT_SET(hadc->Instance->CFGR1, ADC_CFGR1_DMAEN) )
1561#else
1562 if (hadc->Init.Overrun == ADC_OVR_DATA_PRESERVED)
1563#endif
1564 {
1565 /* Set ADC error code to overrun */
1566 SET_BIT(hadc->ErrorCode, HAL_ADC_ERROR_OVR);
1567
1568 /* Clear ADC overrun flag */
1569 __HAL_ADC_CLEAR_FLAG(hadc, ADC_FLAG_OVR);
1570
1571#if (USE_HAL_ADC_REGISTER_CALLBACKS == 1)
1572 hadc->ErrorCallback(hadc);
1573#else
1574 HAL_ADC_ErrorCallback(hadc);
1575#endif /* USE_HAL_ADC_REGISTER_CALLBACKS */
1576 }
1577
1578 /* Clear the Overrun flag */
1579 __HAL_ADC_CLEAR_FLAG(hadc, ADC_FLAG_OVR);
1580 }
