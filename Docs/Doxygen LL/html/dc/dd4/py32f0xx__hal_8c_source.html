146HAL_StatusTypeDef HAL_Init(void)
147{

150 /* Configure Flash prefetch, Instruction cache */
151 /* Default configuration at reset is: */
152 /* - Prefetch disabled */
153 /* - Instruction cache enabled */
154
155 /* Use SysTick as time base source and configure 1ms tick (default clock after Reset is HSI) */

158 status = HAL_ERROR;
159 }
160 else

235__weak HAL_StatusTypeDef HAL_InitTick(uint32_t TickPriority)
236{
237 HAL_StatusTypeDef status = HAL_OK;
238
239 if (uwTickFreq != 0U)
240 {
241 /*Configure the SysTick to have interrupt in 1ms time basis*/
242 if (HAL_SYSTICK_Config(SystemCoreClock / (1000U /uwTickFreq)) == 0U)
243 {
244 /* Configure the SysTick IRQ priority */
245 if (TickPriority < (1UL << __NVIC_PRIO_BITS))
246 {
247 HAL_NVIC_SetPriority(SysTick_IRQn, TickPriority, 0U);
248 uwTickPrio = TickPriority;
249 }
250 else
251 {
252 status = HAL_ERROR;
253 }
254 }
255 else
256 {
257 status = HAL_ERROR;
258 }
259 }
260 else
261 {
262 status = HAL_ERROR;
263 }
264
265 /* Return function status */
266 return status;
267}

331HAL_StatusTypeDef HAL_SetTickFreq(uint32_t Freq)
332{
333 HAL_StatusTypeDef status = HAL_OK;
334 assert_param(IS_TICKFREQ(Freq));
335
336 if (uwTickFreq != Freq)
337 {
338 /* Apply the new tick Freq */
