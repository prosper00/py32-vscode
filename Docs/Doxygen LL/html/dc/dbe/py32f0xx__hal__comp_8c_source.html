234HAL_StatusTypeDef HAL_COMP_Init(COMP_HandleTypeDef *hcomp)
235{
236 uint32_t tmp_csr;
237 uint32_t exti_line;
238 uint32_t comp_voltage_scaler_initialized; /* Value "0" if comparator voltage scaler is not initialized */
239 __IO uint32_t wait_loop_index = 0UL;
240 HAL_StatusTypeDef status = HAL_OK;
241
242 /* Check the COMP handle allocation and lock status */
243 if(hcomp == NULL)
244 {
245 status = HAL_ERROR;
246 }
247 else if(__HAL_COMP_IS_LOCKED(hcomp))
248 {
249 status = HAL_ERROR;
250 }
251 else
252 {
253 /* Check the parameters */
254 assert_param(IS_COMP_ALL_INSTANCE(hcomp->Instance));
255 assert_param(IS_COMP_INPUT_PLUS(hcomp->Instance, hcomp->Init.InputPlus));
256 assert_param(IS_COMP_INPUT_MINUS(hcomp->Instance, hcomp->Init.InputMinus));
257 assert_param(IS_COMP_OUTPUTPOL(hcomp->Init.OutputPol));
258 assert_param(IS_COMP_POWERMODE(hcomp->Init.Mode));
259 assert_param(IS_COMP_HYSTERESIS(hcomp->Init.Hysteresis));
260 assert_param(IS_COMP_TRIGGERMODE(hcomp->Init.TriggerMode));
261 assert_param(IS_COMP_WINDOWMODE(hcomp->Init.WindowMode));
262
263 if(hcomp->State == HAL_COMP_STATE_RESET)
264 {
265 /* Allocate lock resource and initialize it */
266 hcomp->Lock = HAL_UNLOCKED;
267
268 /* Set COMP error code to none */
269 COMP_CLEAR_ERRORCODE(hcomp);
270
271
272#if (USE_HAL_COMP_REGISTER_CALLBACKS == 1)
273 /* Init the COMP Callback settings */
274 hcomp->TriggerCallback = HAL_COMP_TriggerCallback; /* Legacy weak callback */
275
276 if (hcomp->MspInitCallback == NULL)
277 {
278 hcomp->MspInitCallback = HAL_COMP_MspInit; /* Legacy weak MspInit */
279 }
280
281 /* Init the low level hardware */
282 /* Note: Internal control clock of the comparators must */
283 /* be enabled in "HAL_COMP_MspInit()" */
284 /* using "__HAL_RCC_SYSCFG_CLK_ENABLE()". */
285 hcomp->MspInitCallback(hcomp);
286#else
287 /* Init the low level hardware */
288 /* Note: Internal control clock of the comparators must */
289 /* be enabled in "HAL_COMP_MspInit()" */
290 /* using "__HAL_RCC_SYSCFG_CLK_ENABLE()". */
291 HAL_COMP_MspInit(hcomp);
292#endif /* USE_HAL_COMP_REGISTER_CALLBACKS */
293 }
294
295 /* Memorize voltage scaler state before initialization */
296 comp_voltage_scaler_initialized = READ_BIT(hcomp->Instance->CSR, (COMP_CSR_INMSEL_1 | COMP_CSR_INMSEL_0));
297
298 /* Set COMP parameters */
299 tmp_csr = ( hcomp->Init.InputMinus
300 | hcomp->Init.InputPlus
301 | hcomp->Init.OutputPol
302 | hcomp->Init.Mode
303 );
304
305 /* Set parameters in COMP register */
306 /* Note: Update all bits except read-only, lock and enable bits */
307 MODIFY_REG(hcomp->Instance->CSR,
308 COMP_CSR_LOCK | COMP_CSR_COMP_OUT | COMP_CSR_PWRMODE |
309 COMP_CSR_POLARITY | COMP_CSR_WINMODE | COMP_CSR_INPSEL |
310 COMP_CSR_INMSEL | COMP_CSR_EN,
311 tmp_csr
312 );
313
314 /* Set digital filter */
315 if (hcomp->Init.DigitalFilter == 0)
316 {
317 /* Disable digital filter */
318 CLEAR_BIT(hcomp->Instance->FR, COMP_FR_FLTEN);
319 }else
320 {
321 WRITE_REG(hcomp->Instance->FR, (COMP_FR_FLTEN | ((hcomp->Init.DigitalFilter) << COMP_FR_FLTCNT_Pos)));
322 }
323
324 /* Set Vrefint Scaler Enable */
325 if ((hcomp->Init.InputMinus == COMP_INPUT_MINUS_1_4VREFINT) || (hcomp->Init.InputMinus == COMP_INPUT_MINUS_1_2VREFINT) || \
326 (hcomp->Init.InputMinus == COMP_INPUT_MINUS_3_4VREFINT) || (hcomp->Init.InputMinus == COMP_INPUT_MINUS_VREFINT))
327 {
328 SET_BIT(COMP12_COMMON->CSR_ODD, COMP_CSR_SCALER_EN);
329 }
330
331 /* Set comparator hysteresis mode */
332 MODIFY_REG(COMP12_COMMON->CSR_ODD, COMP_CSR_HYST, hcomp->Init.Hysteresis);
333
334 /* Set window mode */
335 /* Note: Window mode bit is located into 1 out of the 2 pairs of COMP */
336 /* instances. Therefore, this function can update another COMP */
337 /* instance that the one currently selected. */
338 if(hcomp->Init.WindowMode == COMP_WINDOWMODE_COMP1_INPUT_PLUS_COMMON)
339 {
340 CLEAR_BIT(COMP12_COMMON->CSR_ODD, COMP_CSR_WINMODE);
341 SET_BIT(COMP12_COMMON->CSR_EVEN, COMP_CSR_WINMODE);
342 }
343 else if(hcomp->Init.WindowMode == COMP_WINDOWMODE_COMP2_INPUT_PLUS_COMMON)
344 {
345 SET_BIT(COMP12_COMMON->CSR_ODD, COMP_CSR_WINMODE);
346 CLEAR_BIT(COMP12_COMMON->CSR_EVEN, COMP_CSR_WINMODE);
347 }
348 else
349 {
350 CLEAR_BIT(COMP12_COMMON->CSR_ODD, COMP_CSR_WINMODE);
351 CLEAR_BIT(COMP12_COMMON->CSR_EVEN, COMP_CSR_WINMODE);
352 }
353
354 /* Delay for COMP scaler bridge voltage stabilization */
355 /* Apply the delay if voltage scaler bridge is required and not already enabled */
356 if ((READ_BIT(hcomp->Instance->CSR, (COMP_CSR_INMSEL_1 | COMP_CSR_INMSEL_0)) != 0UL) &&
357 (comp_voltage_scaler_initialized == 0UL) )
358 {
359 /* Wait loop initialization and execution */
360 /* Note: Variable divided by 2 to compensate partially */
361 /* CPU processing cycles, scaling in us split to not */
362 /* exceed 32 bits register capacity and handle low frequency. */
363 wait_loop_index = ((COMP_DELAY_VOLTAGE_SCALER_STAB_US / 10UL) * (SystemCoreClock / (100000UL * 2UL)));
364 while(wait_loop_index != 0UL)
365 {
366 wait_loop_index--;
367 }
368 }
369
370 /* Get the EXTI line corresponding to the selected COMP instance */
371 exti_line = COMP_GET_EXTI_LINE(hcomp->Instance);
372
373 /* Manage EXTI settings */
374 if((hcomp->Init.TriggerMode & (COMP_EXTI_IT | COMP_EXTI_EVENT)) != 0UL)
375 {
376 /* Configure EXTI rising edge */
377 if((hcomp->Init.TriggerMode & COMP_EXTI_RISING) != 0UL)
378 {
379 LL_EXTI_EnableRisingTrig(exti_line);
380 }
381 else
382 {
383 LL_EXTI_DisableRisingTrig(exti_line);
384 }
385
386 /* Configure EXTI falling edge */
387 if((hcomp->Init.TriggerMode & COMP_EXTI_FALLING) != 0UL)
388 {
389 LL_EXTI_EnableFallingTrig(exti_line);
390 }
391 else
392 {
393 LL_EXTI_DisableFallingTrig(exti_line);
394 }
395
396 /* Clear COMP EXTI pending bit (if any) */
397 LL_EXTI_ClearFlag(exti_line);
398
399 /* Configure EXTI event mode */
400 if((hcomp->Init.TriggerMode & COMP_EXTI_EVENT) != 0UL)
401 {
402 LL_EXTI_EnableEvent(exti_line);
403 }
404 else
405 {
406 LL_EXTI_DisableEvent(exti_line);
407 }
408
409 /* Configure EXTI interrupt mode */
410 if((hcomp->Init.TriggerMode & COMP_EXTI_IT) != 0UL)
411 {
412 LL_EXTI_EnableIT(exti_line);
413 }
414 else
415 {
416 LL_EXTI_DisableIT(exti_line);
417 }
418 }
419 else
420 {
421 /* Disable EXTI event mode */
422 LL_EXTI_DisableEvent(exti_line);
423
424 /* Disable EXTI interrupt mode */
425 LL_EXTI_DisableIT(exti_line);
426 }
427
428 /* Set HAL COMP handle state */
429 /* Note: Transition from state reset to state ready, */
430 /* otherwise (coming from state ready or busy) no state update. */
431 if (hcomp->State == HAL_COMP_STATE_RESET)
432 {
433 hcomp->State = HAL_COMP_STATE_READY;
434 }
435 }
436
437 return status;
438}

448HAL_StatusTypeDef HAL_COMP_DeInit(COMP_HandleTypeDef *hcomp)
449{
450 HAL_StatusTypeDef status = HAL_OK;
451
452 /* Check the COMP handle allocation and lock status */
453 if(hcomp == NULL)
454 {
455 status = HAL_ERROR;
456 }
457 else if(__HAL_COMP_IS_LOCKED(hcomp))
458 {
459 status = HAL_ERROR;
460 }
461 else
462 {
463 /* Check the parameter */
464 assert_param(IS_COMP_ALL_INSTANCE(hcomp->Instance));
465
466 /* Set COMP_CSR register to reset value */
467 WRITE_REG(hcomp->Instance->CSR, 0x00000000UL);
468
469#if (USE_HAL_COMP_REGISTER_CALLBACKS == 1)
470 if (hcomp->MspDeInitCallback == NULL)

473 }
474
475 /* DeInit the low level hardware: GPIO, RCC clock, NVIC */
476 hcomp->MspDeInitCallback(hcomp);
477#else
478 /* DeInit the low level hardware: GPIO, RCC clock, NVIC */
479 HAL_COMP_MspDeInit(hcomp);
480#endif /* USE_HAL_COMP_REGISTER_CALLBACKS */
481
482 /* Set HAL COMP handle state */
483 hcomp->State = HAL_COMP_STATE_RESET;
484
485 /* Release Lock */
486 __HAL_UNLOCK(hcomp);
487 }
488
489 return status;

800void HAL_COMP_IRQHandler(COMP_HandleTypeDef *hcomp)
801{
802 /* Get the EXTI line corresponding to the selected COMP instance */
803 uint32_t exti_line = COMP_GET_EXTI_LINE(hcomp->Instance);
804 uint32_t comparator_window_mode_odd = READ_BIT(COMP12_COMMON->CSR_ODD, COMP_CSR_WINMODE);
805 uint32_t comparator_window_mode_even = READ_BIT(COMP12_COMMON->CSR_EVEN, COMP_CSR_WINMODE);
806
807 /* Check COMP EXTI flag */
808 if(LL_EXTI_IsActiveFlag(exti_line) != 0UL)
809 {
810 /* Check whether comparator is in independent or window mode */
811 if( (comparator_window_mode_odd != 0UL)
812 || (comparator_window_mode_even != 0UL))
813 {
814 /* Clear COMP EXTI line pending bit of the pair of comparators */
815 /* in window mode. */
816 /* Note: Pair of comparators in window mode can both trig IRQ when */
817 /* input voltage is changing from "out of window" area */
818 /* (low or high ) to the other "out of window" area (high or low).*/
819 /* Both flags must be cleared to call comparator trigger */
820 /* callback is called once. */
821 LL_EXTI_ClearFlag((COMP_EXTI_LINE_COMP1 | COMP_EXTI_LINE_COMP2));
822 }
823 else
824 {
825 /* Clear COMP EXTI line pending bit */
826 LL_EXTI_ClearFlag(exti_line);
827 }
828
829 /* COMP trigger user callback */
830#if (USE_HAL_COMP_REGISTER_CALLBACKS == 1)
831 hcomp->TriggerCallback(hcomp);
832#else
833 HAL_COMP_TriggerCallback(hcomp);
834#endif /* USE_HAL_COMP_REGISTER_CALLBACKS */
835 }
836 else
837 {
838 /* nothing to do */
839 }
840}

868HAL_StatusTypeDef HAL_COMP_Lock(COMP_HandleTypeDef *hcomp)
869{
870 HAL_StatusTypeDef status = HAL_OK;
871
872 /* Check the COMP handle allocation and lock status */
873 if(hcomp == NULL)
874 {
875 status = HAL_ERROR;
876 }
877 else if(__HAL_COMP_IS_LOCKED(hcomp))
878 {
879 status = HAL_ERROR;
880 }
881 else
882 {
883 /* Check the parameter */
884 assert_param(IS_COMP_ALL_INSTANCE(hcomp->Instance));
885
886 /* Set HAL COMP handle state */
887 switch(hcomp->State)
888 {
889 case HAL_COMP_STATE_RESET:
890 hcomp->State = HAL_COMP_STATE_RESET_LOCKED;
891 break;
892 case HAL_COMP_STATE_READY:
893 hcomp->State = HAL_COMP_STATE_READY_LOCKED;
894 break;
895 default: /* HAL_COMP_STATE_BUSY */
896 hcomp->State = HAL_COMP_STATE_BUSY_LOCKED;
897 break;
898 }
899 }
900
901 if(status == HAL_OK)
902 {
903 /* Set the lock bit corresponding to selected comparator */
904 __HAL_COMP_LOCK(hcomp);
905 }
906
907 return status;
908}
