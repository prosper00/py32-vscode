176void HAL_GPIO_Init(GPIO_TypeDef *GPIOx, GPIO_InitTypeDef *GPIO_Init)
177{
178 uint32_t position = 0x00u;
179 uint32_t iocurrent;
180 uint32_t temp;
181
182 /* Check the parameters */
183 assert_param(IS_GPIO_ALL_INSTANCE(GPIOx));
184 assert_param(IS_GPIO_PIN(GPIO_Init->Pin));
185 assert_param(IS_GPIO_MODE(GPIO_Init->Mode));
186 assert_param(IS_GPIO_PULL(GPIO_Init->Pull));
187
188 /* Configure the port pins */
189 while (((GPIO_Init->Pin) >> position) != 0x00u)
190 {
191 /* Get current io position */
192 iocurrent = (GPIO_Init->Pin) & (1uL << position);
193
194 if (iocurrent != 0x00u)
195 {
196 /*--------------------- GPIO Mode Configuration ------------------------*/
197 /* In case of Alternate function mode selection */
198 if ((GPIO_Init->Mode == GPIO_MODE_AF_PP) || (GPIO_Init->Mode == GPIO_MODE_AF_OD))
199 {
200 /* Check the Alternate function parameters */
201 assert_param(IS_GPIO_AF_INSTANCE(GPIOx));
202 assert_param(IS_GPIO_AF(GPIO_Init->Alternate));
203
204 /* Configure Alternate function mapped with the current IO */
205 temp = GPIOx->AFR[position >> 3u];
206 temp &= ~(0xFu << ((position & 0x07u) * 4u));
207 temp |= ((GPIO_Init->Alternate) << ((position & 0x07u) * 4u));
208 GPIOx->AFR[position >> 3u] = temp;
209 }
210
211 /* Configure IO Direction mode (Input, Output, Alternate or Analog) */
212 temp = GPIOx->MODER;
213 temp &= ~(GPIO_MODER_MODE0 << (position * 2u));
214 temp |= ((GPIO_Init->Mode & GPIO_MODE) << (position * 2u));
215 GPIOx->MODER = temp;
216
217 /* In case of Output or Alternate function mode selection */
218 if ((GPIO_Init->Mode == GPIO_MODE_OUTPUT_PP) || (GPIO_Init->Mode == GPIO_MODE_AF_PP) ||
219 (GPIO_Init->Mode == GPIO_MODE_OUTPUT_OD) || (GPIO_Init->Mode == GPIO_MODE_AF_OD))
220 {
221 /* Check the Speed parameter */
222 assert_param(IS_GPIO_SPEED(GPIO_Init->Speed));
223 /* Configure the IO Speed */
224 temp = GPIOx->OSPEEDR;
225 temp &= ~(GPIO_OSPEEDR_OSPEED0 << (position * 2u));
226 temp |= (GPIO_Init->Speed << (position * 2u));
227 GPIOx->OSPEEDR = temp;
228
229 /* Configure the IO Output Type */
230 temp = GPIOx->OTYPER;
231 temp &= ~(GPIO_OTYPER_OT0 << position) ;
232 temp |= (((GPIO_Init->Mode & GPIO_OUTPUT_TYPE) >> 4u) << position);
233 GPIOx->OTYPER = temp;
234 }
235
236 /* Activate the Pull-up or Pull down resistor for the current IO */
237 temp = GPIOx->PUPDR;
238 temp &= ~(GPIO_PUPDR_PUPD0 << (position * 2u));
239 temp |= ((GPIO_Init->Pull) << (position * 2u));
240 GPIOx->PUPDR = temp;
241
242 /*--------------------- EXTI Mode Configuration ------------------------*/
243 /* Configure the External Interrupt or event for the current IO */
244 if ((GPIO_Init->Mode & EXTI_MODE) == EXTI_MODE)
245 {
246 temp = EXTI->EXTICR[position >> 2u];
247 temp &= ~(0x0FuL << (8u * (position & 0x03u)));
248 temp |= (GPIO_GET_INDEX(GPIOx) << (8u * (position & 0x03u)));
249 EXTI->EXTICR[position >> 2u] = temp;
250
251 /* Clear EXTI line configuration */
252 temp = EXTI->IMR;
253 temp &= ~(iocurrent);
254 if ((GPIO_Init->Mode & GPIO_MODE_IT) == GPIO_MODE_IT)
255 {
256 temp |= iocurrent;
257 }
258 EXTI->IMR = temp;
259
260 temp = EXTI->EMR;
261 temp &= ~(iocurrent);
262 if ((GPIO_Init->Mode & GPIO_MODE_EVT) == GPIO_MODE_EVT)
263 {
264 temp |= iocurrent;
265 }
266 EXTI->EMR = temp;
267
268 /* Clear Rising Falling edge configuration */
269 temp = EXTI->RTSR;
270 temp &= ~(iocurrent);
271 if ((GPIO_Init->Mode & RISING_EDGE) == RISING_EDGE)
272 {
273 temp |= iocurrent;
274 }
275 EXTI->RTSR = temp;
276
277 temp = EXTI->FTSR;
278 temp &= ~(iocurrent);
279 if ((GPIO_Init->Mode & FALLING_EDGE) == FALLING_EDGE)
280 {
281 temp |= iocurrent;
282 }
283 EXTI->FTSR = temp;
284 }
285 }
286
287 position++;
288 }

298void HAL_GPIO_DeInit(GPIO_TypeDef *GPIOx, uint32_t GPIO_Pin)
299{
300 uint32_t position = 0x00u;
301 uint32_t iocurrent;
302 uint32_t tmp;
303
304 /* Check the parameters */
305 assert_param(IS_GPIO_ALL_INSTANCE(GPIOx));
306 assert_param(IS_GPIO_PIN(GPIO_Pin));
307
308 /* Configure the port pins */
309 while ((GPIO_Pin >> position) != 0x00u)
310 {
311 /* Get current io position */
312 iocurrent = (GPIO_Pin) & (1uL << position);
313
314 if (iocurrent != 0x00u)
315 {
316 /*------------------------- EXTI Mode Configuration --------------------*/
317 /* Clear the External Interrupt or Event for the current IO */
318
319 tmp = EXTI->EXTICR[position >> 2u];
320 tmp &= (0x0FuL << (8u * (position & 0x03u)));
321 if (tmp == (GPIO_GET_INDEX(GPIOx) << (8u * (position & 0x03u))))
322 {
323 /* Clear EXTI line configuration */
324 EXTI->IMR &= ~(iocurrent);
325 EXTI->EMR &= ~(iocurrent);
326
327 /* Clear Rising Falling edge configuration */
328 EXTI->RTSR &= ~(iocurrent);
329 EXTI->FTSR &= ~(iocurrent);
330
331 tmp = 0x0FuL << (8u * (position & 0x03u));
332 EXTI->EXTICR[position >> 2u] &= ~tmp;
333 }
334
335 /*------------------------- GPIO Mode Configuration --------------------*/
336 /* Configure IO in Analog Mode */
337 GPIOx->MODER |= (GPIO_MODER_MODE0 << (position * 2u));
338
339 /* Configure the default Alternate Function in current IO */
340 GPIOx->AFR[position >> 3u] &= ~(0xFu << ((position & 0x07u) * 4u)) ;
341
342 /* Configure the default value for IO Speed */
343 GPIOx->OSPEEDR &= ~(GPIO_OSPEEDR_OSPEED0 << (position * 2u));
344
345 /* Configure the default value IO Output Type */
346 GPIOx->OTYPER &= ~(GPIO_OTYPER_OT0 << position) ;
347
348 /* Deactivate the Pull-up and Pull-down resistor for the current IO */
349 GPIOx->PUPDR &= ~(GPIO_PUPDR_PUPD0 << (position * 2u));
350 }
351
352 position++;
353 }

460HAL_StatusTypeDef HAL_GPIO_LockPin(GPIO_TypeDef *GPIOx, uint16_t GPIO_Pin)
461{
462 __IO uint32_t tmp = GPIO_LCKR_LCKK;
463
464 /* Check the parameters */
465 assert_param(IS_GPIO_LOCK_INSTANCE(GPIOx));
466 assert_param(IS_GPIO_PIN(GPIO_Pin));
467
468 /* Apply lock key write sequence */
469 tmp |= GPIO_Pin;
470 /* Set LCKx bit(s): LCKK='1' + LCK[15-0] */
471 GPIOx->LCKR = tmp;
472 /* Reset LCKx bit(s): LCKK='0' + LCK[15-0] */
473 GPIOx->LCKR = GPIO_Pin;
474 /* Set LCKx bit(s): LCKK='1' + LCK[15-0] */
475 GPIOx->LCKR = tmp;
476 /* Read LCKK register. This read is mandatory to complete key lock sequence */
477 tmp = GPIOx->LCKR;
478
479 /* read again in order to confirm lock is active */
480 if ((GPIOx->LCKR & __GPIO_LOCK_KEY_Msk) != 0x00u)
481 {
482 return HAL_OK;
483 }

486 return HAL_ERROR;
